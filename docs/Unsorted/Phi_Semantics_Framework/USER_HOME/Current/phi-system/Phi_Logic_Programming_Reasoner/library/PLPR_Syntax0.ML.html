  (*If the rule contains multiple guards, aggregate them into one guard.
    If there is any non-guard antecedent before a guard, mark the antecedent as a guard.*)

        deep: if true, rulify by Phi_Conv.iso_rulify_conv, pure connectives embedded into HOL
        number of antecedents: ~1 for exhaustively all, Thm.nprems_of + 1 means to rulify the conclusion
        returns how many new antecedent are introduced.*)

val contract_leading_antecedents : bool (*deep*) -> int -> Proof.context -> thm -> thm
      (*the int: how many leading antecedents are considered.*)

val dest_ant_sequence : term -> term list

val elim_ant_sequence : thm -> thm list

fun cond_can_merge \<^const_name>‚ÄπMODE_GUARD‚Ä∫ = true
  | cond_can_merge \<^const_name>‚Äπdefault‚Ä∫ = true
  | cond_can_merge \<^const_name>‚ÄπMODE_SAFE‚Ä∫ = true
  | cond_can_merge _ = false

val merge_conditions =
  let fun need_merge (Const(\<^const_name>‚ÄπPure.imp‚Ä∫, _)
                $ (Const(\<^const_name>‚ÄπTrueprop‚Ä∫, _) $ (Const(\<^const_name>‚ÄπPremise‚Ä∫, _) $ Const(m1, _) $ _))
                $ (Const(\<^const_name>‚ÄπPure.imp‚Ä∫, _)
                $ (Const(\<^const_name>‚ÄπTrueprop‚Ä∫, _) $ (Const(\<^const_name>‚ÄπPremise‚Ä∫, _) $ Const(m2, _) $ _))
                $ _))
            = cond_can_merge m1 andalso m1 = m2
        | need_merge _ = false

      fun qchk tm =
        need_merge tm orelse (case tm of Const(\<^const_name>‚ÄπPure.imp‚Ä∫, _) $ _ $ X => qchk X
                                       | _ => false)

      fun merge ctm =
        if need_merge (Thm.term_of ctm)
        then Conv.rewr_conv @{thm' contract_obligations} ctm
        else Conv.all_conv ctm

      fun conv ctm =
        case Thm.term_of ctm
          of Const(\<^const_name>‚ÄπPure.imp‚Ä∫, _) $ _ $ _ => (Conv.arg_conv conv then_conv merge) ctm
           | _ => Conv.all_conv ctm

   in fn th => if qchk (Thm.prop_of th) then Conv.fconv_rule conv th else th
  end

val guard_atomize = [
  @{lemma' ‚Äπ(P ‚üπ ùóÄùóéùñ∫ùóãùñΩ Q) ‚â° Trueprop (ùóÄùóéùñ∫ùóãùñΩ (P ‚ü∂ Q))‚Ä∫
       by (simp add: ùóãGuard_def atomize_imp)},
  @{lemma' ‚Äπ(‚ãÄx. ùóÄùóéùñ∫ùóãùñΩ Q x) ‚â° Trueprop (ùóÄùóéùñ∫ùóãùñΩ (‚àÄx. Q x))‚Ä∫
       by (simp add: ùóãGuard_def atomize_all)}
]

fun guardize {merge_cond} ctxt =
  let fun conv guard_unfold ctm =
        case Thm.term_of ctm
          of Const(\<^const_name>‚ÄπTrueprop‚Ä∫, _) $ P =>
            (case P of Const(\<^const_name>‚ÄπùóãGuard‚Ä∫, _) $ _ => Conv.all_conv ctm
                     | _ => Conv.arg_conv (Conv.rewr_conv (
                              Thm.transfer'' ctxt @{thm' ùóãGuard_def[symmetric]})) ctm)
           | _ =>
             let val ctxt' = Context.proof_of ctxt
              in ((if guard_unfold
                   then Raw_Simplifier.rewrite ctxt' true guard_atomize
                   else Conv.all_conv)
              then_conv Phi_Conv.atomize_conv (Context.proof_of ctxt)
              then_conv conv guard_unfold) ctm
             end

      fun can_merge_cond (Const(\<^const_name>‚ÄπPure.imp‚Ä∫, _)
                $ (Const(\<^const_name>‚ÄπTrueprop‚Ä∫, _) $ (Const(\<^const_name>‚ÄπùóãGuard‚Ä∫, _) $ (Const(\<^const_name>‚ÄπPremise‚Ä∫, _) $ Const(m1, _) $ _)))
                $ (Const(\<^const_name>‚ÄπPure.imp‚Ä∫, _)
                $ (Const(\<^const_name>‚ÄπTrueprop‚Ä∫, _) $ (Const(\<^const_name>‚ÄπùóãGuard‚Ä∫, _) $ (Const(\<^const_name>‚ÄπPremise‚Ä∫, _) $ Const(m2, _) $ _)))
                $ _))
            = merge_cond andalso cond_can_merge m1 andalso m1 = m2
        | can_merge_cond _ = false

      fun move_and_merge_ants N ctm =
        case N of 0 => Conv.all_conv ctm
                | 1 => Conv.implies_conv (conv true) Conv.all_conv ctm
                | _ =>
         (Conv.implies_conv (conv true) (move_and_merge_ants (N-1)) then_conv
          (fn ctm => Conv.rewr_conv
               (if can_merge_cond (Thm.term_of ctm)
                then @{lemma ‚Äπ(ùóÄùóéùñ∫ùóãùñΩ (Premise mode' P) ‚üπ ùóÄùóéùñ∫ùóãùñΩ (Premise mode Q) ‚üπ PROP R) ‚â° (ùóÄùóéùñ∫ùóãùñΩ (Premise mode (P ‚àß Q)) ‚üπ PROP R)‚Ä∫
                          by (unfold ùóãGuard_def contract_obligations)}
                else @{lemma ‚Äπ(ùóÄùóéùñ∫ùóãùñΩ P ‚üπ ùóÄùóéùñ∫ùóãùñΩ Q ‚üπ PROP R) ‚â° (ùóÄùóéùñ∫ùóãùñΩ P ‚àß‚á©ùóã Q ‚üπ PROP R)‚Ä∫
                          by (unfold ùóãGuard_def Ant_Seq_imp)}
               ) ctm)) ctm

  in fn N => fn rule =>
      case N of 0 => rule
              | 1 => Conv.fconv_rule (Conv.implies_conv (conv false) Conv.all_conv) rule
              | _ => Conv.fconv_rule (move_and_merge_ants N) rule
 end


fun merge_guards {merge_cond} ctxt rule =
  let fun is_guard (Const(\<^const_name>‚ÄπTrueprop‚Ä∫, _) $ (Const(\<^const_name>‚ÄπùóãGuard‚Ä∫, _) $ _)) = true
        | is_guard (Const(\<^const_name>‚ÄπPure.imp‚Ä∫, _) $ _ $ X) = is_guard X
        | is_guard (Const(\<^const_name>‚ÄπPure.all‚Ä∫, _) $ Abs (_, _, X)) = is_guard X
        | is_guard _ = false
      
      val last_ind = find_index is_guard (rev (Thm.prems_of rule))
   in if last_ind = ~1 then rule
      else guardize {merge_cond=merge_cond} ctxt (Thm.nprems_of rule - last_ind) rule
  end

fun atomize_antecedent deep =
  let fun conv ctxt ctm =
        case Thm.term_of ctm
          of Const(\<^const_name>‚ÄπTrueprop‚Ä∫, _) $ _ => Conv.all_conv ctm
           | Const(\<^const_name>‚ÄπPure.imp‚Ä∫, _) $ _ $ _ =>
              (Conv.combination_conv (Conv.arg_conv (conv ctxt)) (conv ctxt) then_conv
               Conv.rewr_conv @{thm' atomize_imp}) ctm
           | Const(\<^const_name>‚ÄπPure.all‚Ä∫, _) $ _ =>
              (Phi_Conv.meta_all_conv (conv o snd) ctxt then_conv
               Conv.rewr_conv @{thm' atomize_all}) ctm
           | _ => if deep then Phi_Conv.iso_atomize_conv ctxt ctm
                  else Conv.all_conv ctm
   in conv
  end

(*N: numbers of leading antecedents to be rulified at most
  returns: numbers of new antecedent extracted*)
fun rulify_antecedents deep N ctxt thm =
  let fun conv N ctxt ctm =
        if N = 0 then Conv.all_conv ctm else
        case Thm.term_of ctm
          of Const(\<^const_name>‚ÄπTrueprop‚Ä∫, _) $ X =>
            (case X
               of Const(\<^const_name>‚ÄπHOL.All‚Ä∫, _) $ _ =>
                    (Conv.rewr_conv @{thm' atomize_all[symmetric]} then_conv
                     Phi_Conv.meta_all_conv (conv N o snd) ctxt then_conv
                     Conv.try_conv (Conv.rewr_conv @{lemma' ‚Äπ(‚ãÄx. True) ‚â° Trueprop True‚Ä∫
                                                        by (rule, simp) })) ctm
                | Const(\<^const_name>‚ÄπHOL.implies‚Ä∫, _) $ _ $ _ =>
                    (Conv.rewr_conv @{thm' atomize_imp[symmetric]} then_conv
                     Conv.combination_conv (Conv.arg_conv (conv ~1 ctxt)) (conv (N-1) ctxt) then_conv
                     Conv.try_conv (Conv.rewr_conv @{lemma' ‚Äπ(PROP P ‚üπ True) ‚â° Trueprop True‚Ä∫
                                                        by (rule, simp) })) ctm
                | Const(\<^const_name>‚ÄπSet.Ball‚Ä∫, _) $ _ $ _ =>
                    (Phi_Conv.rulify_Ball (conv N) ctxt then_conv
                     Conv.try_conv (Conv.rewr_conv @{lemma' ‚Äπmeta_Ball S (Œª_. Trueprop True) ‚â° Trueprop True‚Ä∫
                                                        by (unfold meta_Ball_def, rule, blast)})) ctm
                | _ => if deep then Phi_Conv.iso_rulify_conv ctxt ctm
                       else Conv.all_conv ctm)
           | Const(\<^const_name>‚ÄπPure.imp‚Ä∫, _) $ Y $ _ =>
               (Conv.combination_conv (Conv.arg_conv (conv ~1 ctxt))
                                      (conv (N-1) ctxt) then_conv
                reduce_constant_True_left) ctm
           | Const(\<^const_name>‚ÄπPure.all‚Ä∫, _) $ _ =>
                  Phi_Conv.meta_all_conv (conv N o snd) ctxt ctm
           | _ => Conv.all_conv ctm

      val num = Unsynchronized.ref 0

      fun rule_of (Const(\<^const_name>‚ÄπTrueprop‚Ä∫, _) $ (Const (\<^const_name>‚ÄπAnt_Seq‚Ä∫, _) $ _ $ _))
            = SOME @{thm' Ant_Seq_I}
        | rule_of (Const(\<^const_name>‚ÄπTrueprop‚Ä∫, _) $ (Const (\<^const_name>‚ÄπHOL.conj‚Ä∫, _) $ _ $ _))
            = SOME @{thm' HOL.conjI}
        | rule_of (Const(\<^const_name>‚ÄπTrueprop‚Ä∫, _) $ Const (\<^const_name>‚ÄπHOL.True‚Ä∫, _))
            = SOME @{thm' TrueI}
        | rule_of _ = NONE
      exception OUT_OF_IDX
      fun pass 0 _ thm = thm
        | pass N i thm =
           (case (rule_of (Logic.strip_assums_concl (Logic.nth_prem (i+1, Thm.prop_of thm)))
                       handle TERM _ => raise OUT_OF_IDX)
              of SOME rule => ( num := !num + 1 ;
                                pass (if N < 0 then N else N+1) i (rule RSN' (ctxt, i+1, thm)) )
               | NONE => thm
                      |> Conv.gconv_rule (conv ~1 ctxt) 1
                      |> pass (N-1) (i+1))
            handle OUT_OF_IDX =>
              let val thm = Conv.fconv_rule (Conv.concl_conv (Thm.nprems_of thm) (conv N ctxt)) thm
               in if i < Thm.nprems_of thm
               then pass N i thm
               else thm
              end
      val thm = pass N 0 thm
   in (!num, thm)
  end

fun contract_leading_antecedents deep N ctxt rule =
  let fun conv N ctxt ctm =
        case N
          of 0 => Conv.all_conv ctm
           | 1 => Conv.implies_conv (atomize_antecedent deep ctxt) Conv.all_conv ctm
           | _ =>(case Thm.term_of ctm
                    of Const(\<^const_name>‚ÄπPure.imp‚Ä∫, _) $ _ $ _ =>
                        (Conv.combination_conv (Conv.arg_conv (atomize_antecedent deep ctxt))
                                               (conv (N-1) ctxt) then_conv
                         Conv.rewr_conv @{thm' Ant_Seq_imp[symmetric]}) ctm
                     | _ => raise CTERM ("contract_leading_antecedents: insufficient antecedents", []))
   in Conv.fconv_rule (conv N ctxt) rule
  end

val ant_seq = \<^Const>‚ÄπAnt_Seq‚Ä∫
fun mk_ant_sequence [] = \<^const>‚ÄπTrue‚Ä∫
  | mk_ant_sequence L = foldr1 (fn (t1,t2) => ant_seq $ t1 $ t2) L

fun dest_ant_sequence tm =
  let fun dest ret (Const(\<^const_name>‚ÄπAnt_Seq‚Ä∫, _) $ A $ B) = dest (dest ret B) A
        | dest ret (Const(\<^const_name>‚ÄπTrue‚Ä∫, _)) = ret
        | dest ret X = X :: ret
   in dest [] tm
  end
fun dest_ant_sequence_or_HOL_conj tm =
  let fun dest ret (Const(\<^const_name>‚ÄπAnt_Seq‚Ä∫, _) $ A $ B) = dest (dest ret B) A
        | dest ret (Const(\<^const_name>‚ÄπHOL.conj‚Ä∫, _) $ A $ B) = dest (dest ret B) A
        | dest ret (Const(\<^const_name>‚ÄπTrue‚Ä∫, _)) = ret
        | dest ret X = X :: ret
   in dest [] tm
  end

fun intr_ant_seq thP thQ =
  let
    val (P, Q) = apply2 (Thm.dest_arg o Thm.cprop_of) (thP, thQ)
      handle CTERM (msg, _) => raise THM (msg, 0, [thP, thQ]);
    val rule = \<^instantiate>‚ÄπP and Q in lemma (open) ‚ÄπP ‚üπ Q ‚üπ P ‚àß‚á©ùóã Q‚Ä∫ by (rule Ant_Seq_I)‚Ä∫
  in Drule.implies_elim_list rule [thP, thQ] end;

fun intro_ant_sequence [] = @{thm' TrueI}
  | intro_ant_sequence L = foldr1 (uncurry intr_ant_seq) L

fun elim_ant_seq thPQ =
  let
    val _ = case Thm.prop_of thPQ
              of Const(\<^const_name>‚ÄπTrueprop‚Ä∫, _) $ (Const(\<^const_name>‚ÄπAnt_Seq‚Ä∫, _) $ _ $ _) => ()
               | _ => raise THM ("elim_ant_seq", 0, [thPQ])
    val (P, Q) = Thm.dest_binop (Thm.dest_arg (Thm.cprop_of thPQ))
      handle CTERM (msg, _) => raise THM (msg, 0, [thPQ]);
    val thP =
      Thm.implies_elim \<^instantiate>‚ÄπP and Q in lemma (open) ‚ÄπP ‚àß‚á©ùóã Q ‚üπ P‚Ä∫ by (unfold Ant_Seq_def, rule conjunct1)‚Ä∫ thPQ;
    val thQ =
      Thm.implies_elim \<^instantiate>‚ÄπP and Q in lemma (open) ‚ÄπP ‚àß‚á©ùóã Q ‚üπ Q‚Ä∫ by (unfold Ant_Seq_def, rule conjunct2)‚Ä∫ thPQ;
  in (thP, thQ) end

fun elim_conj thPQ =
  let
    val _ = case Thm.prop_of thPQ
              of Const(\<^const_name>‚ÄπTrueprop‚Ä∫, _) $ (Const(\<^const_name>‚ÄπHOL.conj‚Ä∫, _) $ _ $ _) => ()
               | _ => raise THM ("elim_conj", 0, [thPQ])
    val (P, Q) = Thm.dest_binop (Thm.dest_arg (Thm.cprop_of thPQ))
      handle CTERM (msg, _) => raise THM (msg, 0, [thPQ]);
    val thP =
      Thm.implies_elim \<^instantiate>‚ÄπP and Q in lemma (open) ‚ÄπP ‚àß Q ‚üπ P‚Ä∫ by (rule conjunct1)‚Ä∫ thPQ;
    val thQ =
      Thm.implies_elim \<^instantiate>‚ÄπP and Q in lemma (open) ‚ÄπP ‚àß Q ‚üπ Q‚Ä∫ by (rule conjunct2)‚Ä∫ thPQ;
  in (thP, thQ) end

fun elim_ant_seq_or_HOL_conj thPQ =
  case Thm.prop_of thPQ
    of _ (*Trueprop*) $ (Const(\<^const_name>‚ÄπHOL.conj‚Ä∫, _) $ _ $ _) => elim_conj thPQ
     | _ (*Trueprop*) $ (Const(\<^const_name>‚ÄπAnt_Seq‚Ä∫, _) $ _ $ _) => elim_ant_seq thPQ
     | _ => raise THM ("elim_ant_seq_or_HOL_conj", 0, [thPQ])

fun elim_ant_sequence th =
  case Thm.prop_of th
    of Const(\<^const_name>‚ÄπTrueprop‚Ä∫, _) $ Const(\<^const_name>‚ÄπTrue‚Ä∫, _) => []
     | _ => let fun elim th =
                  let val (th1, th2) = elim_ant_seq th
                   in elim th1 @ elim th2
                  end handle THM _ => [th]
             in elim th
            end

fun elim_ant_sequence_or_HOL_conj th =
  case Thm.prop_of th
    of Const(\<^const_name>‚ÄπTrueprop‚Ä∫, _) $ Const(\<^const_name>‚ÄπTrue‚Ä∫, _) => []
     | _ => let fun elim th =
                  let val (th1, th2) = elim_ant_seq_or_HOL_conj th
                   in elim th1 @ elim th2
                  end handle THM _ => [th]
             in elim th
            end
