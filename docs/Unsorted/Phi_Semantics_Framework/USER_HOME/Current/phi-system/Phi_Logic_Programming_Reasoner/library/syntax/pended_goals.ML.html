signature PLPR_SYNTAX = sig
include PLPR_SYNTAX

val is_deferable : term -> bool

val non_deferable_subgoals  : term -> int
val non_deferable_subgoals' : thm  -> int

end

structure PLPR_Syntax : PLPR_SYNTAX = struct
open PLPR_Syntax

fun is_deferable (Const (\<^const_name>‹ParamTag›, _) $ _) = true
  | is_deferable (Const (\<^const_name>‹Argument›, _) $ _) = true
  | is_deferable (Const (\<^const_name>‹Do›, _) $ _) = true
  | is_deferable (\<^const>‹Trueprop› $ X) = is_deferable X
  | is_deferable (Const(\<^const_name>‹Pure.all›, _) $ Abs (_,_,X)) = is_deferable X
  | is_deferable (Const(\<^const_name>‹Pure.imp›, _) $ _ $ X) = is_deferable X
  (*| is_deferable (Const (\<^const_name>‹Premise›, _) $ _) = true *)
  | is_deferable _ = false

val non_deferable_subgoals =
  let fun calc ret (Const(\<^const_name>‹Pure.imp›, _) $ G $ C) =
            calc (if is_deferable G then ret else ret+1) C
        | calc ret _ = ret
   in calc 0
  end

fun non_deferable_subgoals' st = non_deferable_subgoals (Thm.prop_of st)

end
