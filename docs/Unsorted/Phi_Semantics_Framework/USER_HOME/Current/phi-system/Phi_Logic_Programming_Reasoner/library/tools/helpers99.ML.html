fun fold_antecedents F =
  let fun fold bvtys (Const(\<^const_name>‚ÄπPure.imp‚Ä∫, _) $ X $ Y) x = fold bvtys Y (fold bvtys X x)
        | fold bvtys (Const(\<^const_name>‚ÄπPure.all‚Ä∫, _) $ Abs (N, Ty, X)) x = fold ((N,Ty)::bvtys) X x
        | fold bvtys (Const(\<^const_name>‚Äπpure_all_embed‚Ä∫, _) $ Abs (N, Ty, X)) x = fold ((N,Ty)::bvtys) X x
        | fold bvtys (Const(\<^const_name>‚Äπmeta_Ball‚Ä∫, _) $ _ $ Abs (N, Ty, X)) x
            = fold ((N,Ty)::bvtys) X x
        | fold bvtys (Const(\<^const_name>‚ÄπùóãGuard‚Ä∫, _) $ X) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‚ÄπAnt_Seq‚Ä∫, _) $ X $ Y) x
            = fold bvtys Y (fold bvtys X x)
        | fold bvtys (Const(\<^const_name>‚ÄπTrueprop‚Ä∫, _) $ X) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‚Äπpure_prop_embed‚Ä∫, _) $ X) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‚ÄπDo‚Ä∫, _) $ X) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‚ÄπDo_embed‚Ä∫, _) $ X) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‚ÄπAction_Tag‚Ä∫, _) $ X $ _) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‚ÄπMeta_Tag‚Ä∫, _) $ X $ _) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‚ÄπNO_SIMP‚Ä∫, _) $ X) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‚ÄπNO_SIMP'‚Ä∫, _) $ X) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‚ÄπWith_IN_Arg‚Ä∫, _) $ X $ _) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‚ÄπWith_Pattern‚Ä∫, _) $ X $ _) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‚ÄπExcept_Pattern‚Ä∫, _) $ X $ _) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‚ÄπHOL.implies‚Ä∫, _) $ X $ Y) x
            = fold bvtys Y (fold bvtys X x)
        | fold bvtys (Const(\<^const_name>‚Äπpure_imp_embed‚Ä∫, _) $ X $ Y) x
            = fold bvtys Y (fold bvtys X x)
        | fold bvtys (Const(\<^const_name>‚Äπpure_conj_embed‚Ä∫, _) $ X $ Y) x
            = fold bvtys Y (fold bvtys X x)
        | fold bvtys (Const(\<^const_name>‚ÄπHOL.conj‚Ä∫, _) $ X $ Y) x

        | fold bvtys (Const(\<^const_name>‚ÄπBranch‚Ä∫, _) $ X $ Y) x

        | fold bvtys (Const(\<^const_name>‚ÄπBranch_embed‚Ä∫, _) $ X $ Y) x

        | fold bvtys (Const(\<^const_name>‚ÄπHOL.disj‚Ä∫, _) $ X $ Y) x

fun app_antecedents F bvs term =
  fold_antecedents (fn bvs => fn term => fn _ => F bvs term) bvs term ()

fun eixsts_antecedent P X = fold_antecedents (fn _ => fn X => fn b => b orelse P X) [] X false
fun forall_antecedent P X = fold_antecedents (fn _ => fn X => fn b => b andalso P X) [] X true
