signature PLPR_SYNTAX = sig
include PLPR_SYNTAX

(*strips all controlling connectives*)
val fold_antecedents : (bvs -> term -> 'a -> 'a)
                    -> bvs -> term -> 'a -> 'a
val concl_of_antecedent : bvs -> term -> bvs * term

val eixsts_antecedent : (term -> bool) -> term -> bool
val forall_antecedent : (term -> bool) -> term -> bool
val app_antecedents : (bvs -> term -> unit)
                    -> bvs -> term -> unit

end

structure PLPR_Syntax : PLPR_SYNTAX = struct
open PLPR_Syntax

val concl_of_antecedent =
  let fun concl bvs (Const(\<^const_name>‹Pure.imp›, _) $ _ $ X) = concl bvs X
        | concl bvs (Const(\<^const_name>‹Pure.all›, _) $ Abs (N, Ty, X)) = concl ((N,Ty)::bvs) X
        | concl bvs (Const(\<^const_name>‹pure_all_embed›, _) $ Abs (N, Ty, X)) = concl ((N,Ty)::bvs) X
        | concl bvs (Const(\<^const_name>‹meta_Ball›, _) $ _ $ Abs (N, Ty, X))
            = concl ((N,Ty)::bvs) X
        | concl bvs (Const(\<^const_name>‹𝗋Guard›, _) $ X) = concl bvs X
        | concl bvs (Const(\<^const_name>‹Trueprop›, _) $ X) = concl bvs X
        | concl bvs (Const(\<^const_name>‹pure_prop_embed›, _) $ X) = concl bvs X
        | concl bvs (Const(\<^const_name>‹Do›, _) $ X) = concl bvs X
        | concl bvs (Const(\<^const_name>‹Do_embed›, _) $ X) = concl bvs X
        | concl bvs (Const(\<^const_name>‹Action_Tag›, _) $ X $ _) = concl bvs X
        | concl bvs (Const(\<^const_name>‹Meta_Tag›, _) $ X $ _) = concl bvs X
        | concl bvs (Const(\<^const_name>‹NO_SIMP›, _) $ X) = concl bvs X
        | concl bvs (Const(\<^const_name>‹NO_SIMP'›, _) $ X) = concl bvs X
        | concl bvs (Const(\<^const_name>‹With_IN_Arg›, _) $ X $ _) = concl bvs X
        | concl bvs (Const(\<^const_name>‹With_Pattern›, _) $ X $ _) = concl bvs X
        | concl bvs (Const(\<^const_name>‹Except_Pattern›, _) $ X $ _) = concl bvs X
        | concl bvs (Const(\<^const_name>‹HOL.implies›, _) $ _ $ X) = concl bvs X
        | concl bvs (Const(\<^const_name>‹Try›, _) $ X) = concl bvs X
        | concl bvs (Const(\<^const_name>‹Attempt_Fallback›, _) $ X) = concl bvs X
        | concl bvs (Const(\<^const_name>‹Memoize›, _) $ X) = concl bvs X
        | concl bvs X = (bvs,X)
   in concl
  end

fun fold_antecedents F =
  let fun fold bvtys (Const(\<^const_name>‹Pure.imp›, _) $ X $ Y) x = fold bvtys Y (fold bvtys X x)
        | fold bvtys (Const(\<^const_name>‹Pure.all›, _) $ Abs (N, Ty, X)) x = fold ((N,Ty)::bvtys) X x
        | fold bvtys (Const(\<^const_name>‹pure_all_embed›, _) $ Abs (N, Ty, X)) x = fold ((N,Ty)::bvtys) X x
        | fold bvtys (Const(\<^const_name>‹meta_Ball›, _) $ _ $ Abs (N, Ty, X)) x
            = fold ((N,Ty)::bvtys) X x
        | fold bvtys (Const(\<^const_name>‹𝗋Guard›, _) $ X) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‹Ant_Seq›, _) $ X $ Y) x
            = fold bvtys Y (fold bvtys X x)
        | fold bvtys (Const(\<^const_name>‹Trueprop›, _) $ X) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‹pure_prop_embed›, _) $ X) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‹Do›, _) $ X) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‹Do_embed›, _) $ X) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‹Action_Tag›, _) $ X $ _) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‹Meta_Tag›, _) $ X $ _) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‹NO_SIMP›, _) $ X) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‹NO_SIMP'›, _) $ X) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‹With_IN_Arg›, _) $ X $ _) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‹With_Pattern›, _) $ X $ _) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‹Except_Pattern›, _) $ X $ _) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‹HOL.implies›, _) $ X $ Y) x
            = fold bvtys Y (fold bvtys X x)
        | fold bvtys (Const(\<^const_name>‹pure_imp_embed›, _) $ X $ Y) x
            = fold bvtys Y (fold bvtys X x)
        | fold bvtys (Const(\<^const_name>‹pure_conj_embed›, _) $ X $ Y) x
            = fold bvtys Y (fold bvtys X x)
        | fold bvtys (Const(\<^const_name>‹HOL.conj›, _) $ X $ Y) x
            = fold bvtys Y (fold bvtys X x)
        | fold bvtys (Const(\<^const_name>‹Branch›, _) $ X $ Y) x
            = fold bvtys Y (fold bvtys X x)
        | fold bvtys (Const(\<^const_name>‹Branch_embed›, _) $ X $ Y) x
            = fold bvtys Y (fold bvtys X x)
        | fold bvtys (Const(\<^const_name>‹HOL.disj›, _) $ X $ Y) x
            = fold bvtys Y (fold bvtys X x)
        | fold bvtys (Const(\<^const_name>‹Orelse_shortcut›, _) $ X $ Y) x
            = fold bvtys Y (fold bvtys X x)
        | fold bvtys (Const(\<^const_name>‹Try›, _) $ X) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‹Attempt_Fallback›, _) $ X) x = fold bvtys X x
        | fold bvtys (Const(\<^const_name>‹Memoize›, _) $ X) x = fold bvtys X x
        | fold bvtys X x = F bvtys X x
   in fold
  end

fun app_antecedents F bvs term =
  fold_antecedents (fn bvs => fn term => fn _ => F bvs term) bvs term ()

fun eixsts_antecedent P X = fold_antecedents (fn _ => fn X => fn b => b orelse P X) [] X false
fun forall_antecedent P X = fold_antecedents (fn _ => fn X => fn b => b andalso P X) [] X true

end
