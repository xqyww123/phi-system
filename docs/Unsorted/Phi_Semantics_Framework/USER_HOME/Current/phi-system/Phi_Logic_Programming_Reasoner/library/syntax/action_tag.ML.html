signature PLPR_SYNTAX = sig

  val dest_tags   :  term ->  term
  val dest_tags_c : cterm -> cterm

  val dest_action_of : string (*the leading constant name*) -> term -> term
  val dest_action_of' :  (term -> bool) -> term -> term * term
  val is_action_of : string (*the leading constant name*) -> term -> bool
  val is_action_of' : (term -> bool) -> term -> bool

  val find_premise : (term -> 'a option) -> thm -> 'a option
  val find_tagged_premise': (term -> bool) -> thm -> (term * term) option
          (*scanning over every premises, compacted forms including ∧, ∧⇩𝗋, ⟶*)
  val find_tagged_premise : string -> thm -> (term * term) option

end

structure PLPR_Syntax : PLPR_SYNTAX = struct

fun dest_tags (Const(\<^const_name>‹Trueprop›, _) $ X) = dest_tags X
  | dest_tags (Const(\<^const_name>‹Action_Tag›, _) $ X $ _) = dest_tags X
  | dest_tags (Const(\<^const_name>‹Meta_Tag›, _) $ X $ _) = dest_tags X
  | dest_tags (Const(\<^const_name>‹𝗋Guard›, _) $ X) = dest_tags X
  | dest_tags X = X

fun dest_tags_c X = 
  case Thm.term_of X
    of Const(\<^const_name>‹Trueprop›, _) $ _ => dest_tags_c (Thm.dest_arg X)
     | Const(\<^const_name>‹Action_Tag›, _) $ _ $ _ => dest_tags_c (Thm.dest_arg1 X)
     | Const(\<^const_name>‹Meta_Tag›, _) $ _ $ _  => dest_tags_c (Thm.dest_arg1 X)
     | Const(\<^const_name>‹𝗋Guard›, _) $ _ => dest_tags_c (Thm.dest_arg X)
     | _ => X

fun is_action_of' P (Const(\<^const_name>‹Action_Tag›, _) $ _ $ A) = P A
  | is_action_of' P (Const(\<^const_name>‹Meta_Tag›, _) $ _ $ A) = P A
  | is_action_of' P (Const(\<^const_name>‹Trueprop›, _) $ X) = is_action_of' P X
  | is_action_of' P (Const(\<^const_name>‹𝗋Guard›, _) $ X) = is_action_of' P X
  | is_action_of' _ _ = false
fun is_action_of name = is_action_of' (fn tm => case Term.head_of tm of Const (N',_) => name = N'
                                        | _ => false)

fun dest_action_of' P (tm as (Const(\<^const_name>‹Action_Tag›, _) $ X $ A))
      = if P A then (X, A) else raise TERM ("PLPR_Syntax.dest_action_of", [tm])
  | dest_action_of' P (tm as (Const(\<^const_name>‹Meta_Tag›, _) $ X $ A))
      = if P A then (X, A) else raise TERM ("PLPR_Syntax.dest_action_of", [tm])
  | dest_action_of' P (Const(\<^const_name>‹Trueprop›, _) $ X) = dest_action_of' P X
  | dest_action_of' P (Const(\<^const_name>‹𝗋Guard›, _) $ X) = dest_action_of' P X
  | dest_action_of' _ tm = raise TERM ("PLPR_Syntax.dest_action_of", [tm])

fun dest_action_of name =
      dest_action_of' (fn tm => case Term.head_of tm of Const (N',_) => name = N'
                        | _ => false) #> fst

fun find_premise P =
  let fun find (Const(\<^const_name>‹Trueprop›, _) $ X) = find X
        | find (Const(\<^const_name>‹𝗋Guard›, _) $ X) = find X
        | find (Const(\<^const_name>‹HOL.conj›, _) $ A $ B) =
            (case find A of NONE => find B | some => some)
        | find (Const(\<^const_name>‹Ant_Seq›, _) $ A $ B) =
            (case find A of NONE => find B | some => some)
        | find (Const(\<^const_name>‹Pure.imp›, _) $ _ $ X) = find X
        | find (Const(\<^const_name>‹Pure.all›, _) $ Abs (_, _, X)) = find X
        | find (Const(\<^const_name>‹HOL.All›, _) $ Abs (_, _, X)) = find X
        | find (Const(\<^const_name>‹HOL.implies›, _) $ _ $ X) = find X
        | find X = P X
   in get_first find o Thm.prems_of
  end

fun find_tagged_premise' P = find_premise (
    fn Const(\<^const_name>‹Action_Tag›, _) $ X $ A => if P A then SOME (A, X) else NONE
     | Const(\<^const_name>‹Meta_Tag›, _) $ X $ A => if P A then SOME (A, X) else NONE
     | _ => NONE
  )

fun find_tagged_premise name =
      find_tagged_premise' (fn tm => case Term.head_of tm of Const (N',_) => name = N'
                             | _ => false)

end
