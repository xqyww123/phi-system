structure Phi_Syntax : PHI_SYNTAX = struct
open Phi_Syntax

val arities_of_connective_locker = Synchronized.var "Phi_Syntax.arities_of_connective" ()
val arities_of_connective = Unsynchronized.ref Symtab.empty

fun register_connective iname_airty =
  Synchronized.change arities_of_connective_locker (fn () => (
      arities_of_connective := Symtab.update iname_airty (!arities_of_connective);
      ()))

fun N_arg_rev ret 0 _ = ret
  | N_arg_rev ret N (X $ a) = N_arg_rev (a::ret) (N-1) X

fun arity_of term =
  case Term.head_of term
    of Const (N, _) => Symtab.lookup (!arities_of_connective) N
     | _ => NONE

fun dest_connective_args term = arity_of term
      |> Option.map (fn arity => N_arg_rev [] arity term)

fun conv_N_args 0 C ctm = Conv.all_conv ctm
  | conv_N_args N C ctm = Conv.combination_conv (conv_N_args (N-1) C) C ctm

fun is_BI_connective (Const(\<^const_name>‹times›, \<^Type>‹fun _ \<^Type>‹fun _ \<^Type>‹BI _›››)) = true
  | is_BI_connective (Const(\<^const_name>‹Subjection›, _)) = true
  | is_BI_connective (Const(\<^const_name>‹ExBI›, _)) = true
  | is_BI_connective (Const(\<^const_name>‹φType›, _)) = true
  | is_BI_connective (Const(\<^const_name>‹plus›, \<^Type>‹fun _ \<^Type>‹fun _ \<^Type>‹BI _›››)) = true
  | is_BI_connective (Const(\<^const_name>‹zero_class.zero›, \<^Type>‹BI _›)) = true
  | is_BI_connective (Const(\<^const_name>‹one_class.one›, \<^Type>‹BI _›)) = true
  | is_BI_connective (Const(\<^const_name>‹top›, \<^Type>‹BI _›)) = true
  | is_BI_connective (Const(\<^const_name>‹bot›, \<^Type>‹BI _›)) = true
  | is_BI_connective (Const(\<^const_name>‹inf›, _)) = true
  | is_BI_connective (Const(\<^const_name>‹REMAINS›, _)) = true
  | is_BI_connective (Const(\<^const_name>‹Mul_Quant›, _)) = true
  | is_BI_connective (Const(\<^const_name>‹AllSet›, _)) = true
  | is_BI_connective (A $ _) = is_BI_connective A
  | is_BI_connective (Const (N, _)) = Symtab.defined (!arities_of_connective) N
  | is_BI_connective _ = false

fun fold_items_of_assertion f =
  let fun fold bvs (Const(\<^const_name>‹times›, \<^Type>‹fun _ \<^Type>‹fun _ \<^Type>‹BI _›››) $ A $ B) x = fold bvs A (fold bvs B x)
        | fold bvs (Const(\<^const_name>‹Subjection›, _) $ X $ _) x = fold bvs X x
        | fold bvs (Const(\<^const_name>‹ExBI›, _) $ X) x = fold bvs X x
        | fold bvs (Abs (N, ty, X)) x = fold ((N,ty)::bvs) X x
        | fold bvs (X as (Const(\<^const_name>‹φType›, _) $ _ $ _)) x = f bvs X x
        | fold bvs (Const(\<^const_name>‹plus›, \<^Type>‹fun _ \<^Type>‹fun _ \<^Type>‹BI _›››) $ A $ B) x = fold bvs A (fold bvs B x)
        | fold _ (Const(\<^const_name>‹top›, \<^Type>‹BI _›)) x = x
        | fold _ (Const(\<^const_name>‹bot›, \<^Type>‹BI _›)) x = x
        | fold _ (Const(\<^const_name>‹zero_class.zero›, \<^Type>‹BI _›)) x = x
        | fold _ (Const(\<^const_name>‹one_class.one›, \<^Type>‹BI _›)) x = x
        | fold bvs (Const(\<^const_name>‹AllSet›, _) $ X) x = fold bvs X x
        | fold bvs (Const(\<^const_name>‹inf›, _) $ A $ B) x = fold bvs A (fold bvs B x)
        | fold bvs (Const(\<^const_name>‹REMAINS›, _) $ X $ Y) x = fold bvs Y (fold bvs X x)
        | fold bvs (Const(\<^const_name>‹Mul_Quant›, _) $ X) x = fold bvs X x
        | fold bvs (Const(\<^const_name>‹φTagA›, _) $ _ $ X) x = fold bvs X x
        | fold bvs (Const(\<^const_name>‹If›, _) $ _ $ X $ Y) x = fold bvs Y (fold bvs X x)
        | fold bvs X x = (case dest_connective_args X
                            of SOME args => Basics.fold (fold bvs) args x
                             | _ => f bvs X x)
   in fold
  end

fun exists_item_of_assertion P =
  let fun exists bvs (Const(\<^const_name>‹times›, \<^Type>‹fun _ \<^Type>‹fun _ \<^Type>‹BI _›››) $ A $ B) = exists bvs B orelse exists bvs A

        | exists bvs (Const(\<^const_name>‹ExBI›, _) $ X) = exists bvs X
        | exists bvs (Abs (N, ty, X)) = exists ((N,ty)::bvs) X
        | exists bvs (X as (Const(\<^const_name>‹φType›, _) $ _ $ _)) = P bvs X
        | exists bvs (Const(\<^const_name>‹plus›, \<^Type>‹fun _ \<^Type>‹fun _ \<^Type>‹BI _›››) $ A $ B) = exists bvs B orelse exists bvs A
        | exists _ (Const(\<^const_name>‹top›, \<^Type>‹BI _›)) = false
        | exists _ (Const(\<^const_name>‹bot›, \<^Type>‹BI _›)) = false

        | exists _ (Const(\<^const_name>‹one_class.one›, \<^Type>‹BI _›)) = false
        | exists bvs (Const(\<^const_name>‹AllSet›, _) $ X) = exists bvs X
        | exists bvs (Const(\<^const_name>‹inf›, _) $ A $ B) = exists bvs B orelse exists bvs A
        | exists bvs (Const(\<^const_name>‹REMAINS›, _) $ X $ Y) = exists bvs X orelse exists bvs Y
        | exists bvs (Const(\<^const_name>‹Mul_Quant›, _) $ X) = exists bvs X

  end

fun forall_item_of_assertion_blv P =
  let fun forall lv bvs (Const(\<^const_name>‹times›, \<^Type>‹fun _ \<^Type>‹fun _ \<^Type>‹BI _›››) $ A $ B) = forall lv bvs B andalso forall lv bvs A
        | forall lv bvs (Const(\<^const_name>‹Subjection›, _) $ X $ _) = forall lv bvs X
        | forall lv bvs (Const(\<^const_name>‹ExBI›, _) $ Abs (N, Ty, X)) = forall (lv + 1) ((N,Ty)::bvs) X
        | forall lv bvs (Const(\<^const_name>‹ExBI›, _) $ X) = forall (lv + 1) bvs (Term.incr_boundvars 1 X $ Bound 0)
        | forall lv bvs (Abs (N, Ty, X)) = forall (lv+1) ((N,Ty)::bvs) X
        | forall lv bvs (X as (Const(\<^const_name>‹φType›, _) $ _ $ _)) = P (bvs,lv) X
        | forall lv bvs (Const(\<^const_name>‹plus›, \<^Type>‹fun _ \<^Type>‹fun _ \<^Type>‹BI _›››) $ A $ B) = forall lv bvs B andalso forall lv bvs A
        | forall _ _ (Const(\<^const_name>‹top›, \<^Type>‹BI _›)) = true
        | forall _ _ (Const(\<^const_name>‹bot›, \<^Type>‹BI _›)) = true
        | forall _ _ (Const(\<^const_name>‹zero_class.zero›, \<^Type>‹BI _›)) = true
        | forall _ _ (Const(\<^const_name>‹one_class.one›, \<^Type>‹BI _›)) = true
        | forall lv bvs (Const(\<^const_name>‹AllSet›, _) $ Abs (N, Ty, X)) = forall (lv + 1) ((N,Ty)::bvs) X
        | forall lv bvs (Const(\<^const_name>‹AllSet›, _) $ X) = forall (lv + 1) bvs (Term.incr_boundvars 1 X $ Bound 0)
        | forall lv bvs (Const(\<^const_name>‹inf›, _) $ A $ B) = forall lv bvs B andalso forall lv bvs A
        | forall lv bvs (Const(\<^const_name>‹REMAINS›, _) $ X $ Y) = forall lv bvs X andalso forall lv bvs Y
        | forall lv bvs (Const(\<^const_name>‹Mul_Quant›, _) $ X) = forall lv bvs X
        | forall lv bvs (Const(\<^const_name>‹φTagA›, _) $ _ $ X) = forall lv bvs X
        | forall lv bvs (Const(\<^const_name>‹If›, _) $ _ $ X $ Y) = forall lv bvs X andalso forall lv bvs Y
        | forall lv bvs X = (case dest_connective_args X
                               of SOME args => Library.forall (forall lv bvs) args
                                | _ => P (bvs,lv) X)
   in forall 0
  end

fun conv_items_of_assertion C =
  let fun conv ctxt ctm =
        case Thm.term_of ctm
          of Const(\<^const_name>‹times›, \<^Type>‹fun _ \<^Type>‹fun _ \<^Type>‹BI _›››) $ _ $ _ =>
                Conv.combination_conv (Conv.arg_conv (conv ctxt)) (conv ctxt) ctm
           | Const(\<^const_name>‹Subjection›, _) $ _ $ _ =>
                Conv.arg1_conv (conv ctxt) ctm
           | Const(\<^const_name>‹ExBI›, _) $ _ =>
                Conv.arg_conv (Phi_Conv.abs_conv_eta (conv o snd) ctxt) ctm
           | Const(\<^const_name>‹AllSet›, _) $ _ =>
                Conv.arg_conv (Phi_Conv.abs_conv_eta (conv o snd) ctxt) ctm
           | Const(\<^const_name>‹Mul_Quant›, _) $ _ =>
                Conv.arg_conv (Phi_Conv.abs_conv_eta (conv o snd) ctxt) ctm
           | Const(\<^const_name>‹plus›, \<^Type>‹fun _ \<^Type>‹fun _ \<^Type>‹BI _›››) $ _ $ _ =>
                Conv.combination_conv (Conv.arg_conv (conv ctxt)) (conv ctxt) ctm
           | Const(\<^const_name>‹inf›, _) $ _ $ _ =>
                Conv.combination_conv (Conv.arg_conv (conv ctxt)) (conv ctxt) ctm
           | Const(\<^const_name>‹REMAINS›, _) $ _ $ _ =>
                Conv.combination_conv (Conv.arg_conv (conv ctxt)) (conv ctxt) ctm
           | Const(\<^const_name>‹top›, \<^Type>‹BI _›) => Conv.all_conv ctm
           | Const(\<^const_name>‹bot›, \<^Type>‹BI _›) => Conv.all_conv ctm
           | Const(\<^const_name>‹zero_class.zero›, \<^Type>‹BI _›) => Conv.all_conv ctm
           | Const(\<^const_name>‹one_class.one›, \<^Type>‹BI _›) => Conv.all_conv ctm
           | Const(\<^const_name>‹φTagA›, _) $ _ $ _ => Conv.arg_conv (conv ctxt) ctm
           | Const(\<^const_name>‹If›, _) $ _ $ _ $ _ => (Conv.arg_conv (conv ctxt) then_conv Conv.arg1_conv (conv ctxt)) ctm
           | Abs _ => Conv.abs_conv (conv o snd) ctxt ctm
           | X => (case arity_of X
                     of SOME arity => conv_N_args arity (conv ctxt) ctm
                      | _ => C ctxt ctm)
   in conv
  end

fun conv_all_typings C = conv_items_of_assertion (fn ctxt => fn ctm =>
  case Thm.term_of ctm
    of Const(\<^const_name>‹φType›, _) $ _ $ _ => C ctxt ctm
     | _ => Conv.all_conv ctm)

fun strip_separations (Const (@{const_name Subjection}, _) $ T) = strip_separations T
  | strip_separations (Const (@{const_name ExBI}, _) $ T) = strip_separations T
  | strip_separations (Abs (_, _, T)) = strip_separations T
  | strip_separations T = Phi_Help.strip_binop_r @{const_name times} T

(*the left-most non-branching ending element*)
fun leftmost_sep (Const (\<^const_name>‹times›, _) $ X $ _) = leftmost_sep X
  | leftmost_sep (Const (@{const_name Subjection}, _) $ T) = leftmost_sep T
  | leftmost_sep (Const (@{const_name ExBI}, _) $ T) = leftmost_sep T
  | leftmost_sep (Const (@{const_name AllSet}, _) $ T) = leftmost_sep T
  | leftmost_sep (Const (@{const_name REMAINS}, _) $ X $ _) = leftmost_sep X
  | leftmost_sep (Abs (_, _, T)) = leftmost_sep T
  | leftmost_sep X = X

(*the right-most non-branching ending element*)
fun rightmost_sep (Const (\<^const_name>‹times›, _) $ _ $ X) = rightmost_sep X
  | rightmost_sep (Const (@{const_name Subjection}, _) $ T) = rightmost_sep T
  | rightmost_sep (Const (@{const_name ExBI}, _) $ T) = rightmost_sep T
  | rightmost_sep (Const (@{const_name AllSet}, _) $ T) = rightmost_sep T
  | rightmost_sep (Const (@{const_name REMAINS}, _) $ _ $ X) = rightmost_sep X
  | rightmost_sep (Abs (_, _, T)) = rightmost_sep T
  | rightmost_sep X = X

fun mk_separations model_ty [] = \<^Const>‹one_class.one model_ty›
  | mk_separations model_ty L = foldr1 (fn (A,X) => \<^Const>‹times model_ty› $ A $ X) L

fun mk_phi_type (x,T) =
  let val \<^Type>‹fun ‹a› ‹\<^Type>‹BI ‹b›››› = Phi_Help.safe_fastype_of  T
   in \<^Const>‹φType a b› $ x $ T end

fun mk_phi_type_ty (tm, tx) = \<^Type>‹fun tx \<^Type>‹BI tm››

fun dest_phi_type_assertion (Const(\<^const_name>‹φType›, _) $ x $ T) = (x, T)
  | dest_phi_type_assertion X = raise TERM ("Not a φ-type assertion", [X])

fun dest_phi_type_ty \<^Type>‹fun x \<^Type>‹BI model›› = (x, model)
  | dest_phi_type_ty T = raise TYPE ("dest_phi_type_ty", [T], [])

fun dest_phi_type_ty_c T =
  (dest_phi_type_ty (Thm.typ_of T) ;
   (Thm.dest_ctyp0 T, Thm.dest_ctyp0 (Thm.dest_ctyp1 T)))

fun dest_phi_type_ty' \<^Type>‹fun x ‹\<^Type>‹BI c››› = (x, c)
  | dest_phi_type_ty' \<^Type>‹fun a T› = dest_phi_type_ty' T
  | dest_phi_type_ty' T = raise TYPE ("not a phi-type",[T],[])

fun dest_parameterized_phi_ty' args \<^Type>‹fun x ‹\<^Type>‹BI c››› = (args, x, c)
  | dest_parameterized_phi_ty' args \<^Type>‹fun a T› = dest_parameterized_phi_ty' (a::args) T
  | dest_parameterized_phi_ty' _ T = raise TYPE ("not a phi-type",[T],[])

val dest_parameterized_phi_ty = dest_parameterized_phi_ty' []

fun is_phi_type_ty \<^Type>‹fun _ ‹\<^Type>‹BI _››› = true
  | is_phi_type_ty \<^Type>‹fun _ T› = is_phi_type_ty T
  | is_phi_type_ty _ = false

fun typ_parameter_of_phi_ty bvtys T_term =
  let fun collect_params ret (F' $ T') =
           (case try dest_parameterized_phi_ty (Term.fastype_of1 (bvtys,T'))
              of SOME dest => collect_params ((dest,T') :: ret) F'
               | NONE => collect_params ret F')
        | collect_params ret _  = ret
   in collect_params [] T_term
  end

fun is_nonnull_Type_Opr ty =
  case dest_parameterized_phi_ty ty
    of ([],_,_) => false
     | (args,_,_) => exists (can dest_parameterized_phi_ty) args

fun is_non_parameterized_phityp typ =
  case try dest_parameterized_phi_ty typ
    of SOME ([],_,_) => true
     | _ => false

fun number_of_typ_operators bvtys tm =
  if can dest_phi_type_ty (Term.fastype_of1 (bvtys, tm))
  then let val (_, args) = Term.strip_comb tm
        in fold (fn arg => fn ret => ret + number_of_typ_operators bvtys arg) args 1
       end
  else case tm
    of Abs (_, ty, X) => number_of_typ_operators (ty::bvtys) X
     | _ => 0

fun number_of_typ_operators_in_assertion bvtys term =
      fold_items_of_assertion' (fn bvtys =>
        fn (Const(\<^const_name>‹φType›, _) $ _ $ T) =>
            (fn S => S + number_of_typ_operators bvtys T)
          | _ => I) bvtys term 0
