fun dest_BI_ty (\<^Type>‹BI ty›) = ty
  | dest_BI_ty ty = raise TYPE ("Not of a type of BI assertion", [ty], [])

val dest_transformation   =
         PLPR_Syntax.dest_tags
      #> Phi_Help.dest_triop @{const_name "Transformation"}
val dest_transformation_c =
         PLPR_Syntax.dest_tags_c
      #> Phi_Help.dest_triop_c @{const_name "Transformation"}


fun dest_transformation_typ (Const(\<^const_name>‹Trueprop›, _) $ X) = dest_transformation_typ X
  | dest_transformation_typ (Const(\<^const_name>‹Action_Tag›, _) $ X $ _) = dest_transformation_typ X
  | dest_transformation_typ (Const(\<^const_name>‹Pure.imp›, _) $ X) = dest_transformation_typ X
  | dest_transformation_typ (Const(\<^const_name>‹Transformation›, \<^Type>‹fun \<^Type>‹BI a› _›) $ _ $ _ $ _) = a
  | dest_transformation_typ X = raise TERM ("dest_transformation_typ", [X])

fun is_intro_transformation rule =
  let val (prems, concl) = Logic.strip_horn rule
   in case dest_transformation concl
   of (Var v, _, _) =>
        exists (PLPR_Syntax.eixsts_antecedent (
            fn prem => case try dest_transformation (Logic.strip_assums_concl prem)
                             of SOME (Var v', _, _) => v' = v
                              | _ => false)) prems
    | _ => false
  end

fun is_elim_transformation rule =
  let val (prems, concl) = Logic.strip_horn rule
   in case dest_transformation concl
   of (_, Var v, _) =>
        exists (PLPR_Syntax.eixsts_antecedent (
            fn prem => case try dest_transformation (Logic.strip_assums_concl prem)
                             of SOME (_, Var v', _) => v' = v
                              | _ => false)) prems
    | _ => false
  end

fun pass_ensures_intro_transformation _ (rules, (mode, priority), (pattern, except), guard, ctxt) =
  let fun chk rule = if is_intro_transformation (Thm.prop_of rule) then rule
                     else rule RS' (Context.proof_of ctxt, @{thm' mk_intro_transformation})
   in (map chk rules, (mode, priority), (pattern, except), guard, ctxt)
  end

fun pass_ensures_elim_transformation _ (rules, (mode, priority), (pattern, except), guard, ctxt) =
  let fun chk rule = if is_elim_transformation (Thm.prop_of rule) then rule
                     else rule RS' (Context.proof_of ctxt, @{thm' mk_elim_transformation})
   in (map chk rules, (mode, priority), (pattern, except), guard, ctxt)
  end

fun transformation_conv C_s C_t C_p ctm = (
  dest_transformation_c ctm;
  Phi_Conv.tag_conv (
    Conv.combination_conv (Conv.combination_conv (Conv.combination_conv Conv.all_conv C_s) C_t) C_p) ctm

val enable_auto_chk_and_conv = Config.declare_bool ("enable_auto_chk_and_conv_φToR", ⌂) (K true)
val chk_source_val = Attrib.setup_config_bool \<^binding>‹φchk_source_val› (K true)

fun err ctm = raise CTERM("Fail to classify the Transformation rule automatically into 𝒯𝒫, 𝒯𝒫'", [ctm])
fun err' tm = raise TERM("Fail to classify the Transformation rule automatically into 𝒯𝒫, 𝒯𝒫'", [tm])

fun is_Var X =
  case Term.strip_comb X
    of (Var _, args) => forall (fn Bound _ => true | _ => false) args
     | _ => false

fun conv_ToR_add_tag ctm =
  let datatype kind = ProdTy | Ty | Var | Oth
      fun cond_Ty (Const(\<^const_name>‹φProd'›, _) $ _ $ _) = ProdTy
        | cond_Ty _ = Ty
      fun dest_ty (Const(\<^const_name>‹φType›, _) $ _ $ U) = cond_Ty U
        | dest_ty X = if is_Var X then Var else Oth
      val (X,Y,_) = dest_transformation (Thm.term_of ctm)
   in case (dest_ty X, dest_ty Y)
   of (ProdTy, ProdTy) => Conv.rewr_conv @{thm' Action_Tag_def[where A=‹𝒯𝒫'›, symmetric]} ctm
    | (Var, ProdTy) => Conv.rewr_conv @{thm' Action_Tag_def[where A=‹𝒯𝒫'›, symmetric]} ctm
    | (ProdTy, Var) => Conv.rewr_conv @{thm' Action_Tag_def[where A=‹𝒯𝒫'›, symmetric]} ctm
    | (Oth, _) => Conv.rewr_conv @{thm' Action_Tag_def[where A=‹𝒯𝒫›, symmetric]} ctm
    | (_, Oth) => Conv.rewr_conv @{thm' Action_Tag_def[where A=‹𝒯𝒫›, symmetric]} ctm
    | (Ty, Ty)  => Conv.rewr_conv @{thm' Action_Tag_def[where A=‹𝒯𝒫›, symmetric]} ctm
    | (Ty, Var)  => Conv.rewr_conv @{thm' Action_Tag_def[where A=‹𝒯𝒫›, symmetric]} ctm
    | (Var, Ty)  => Conv.rewr_conv @{thm' Action_Tag_def[where A=‹𝒯𝒫›, symmetric]} ctm
    | (Var, Var)  => Conv.rewr_conv @{thm' Action_Tag_def[where A=‹𝒯𝒫›, symmetric]} ctm
    | _ => err ctm
  end

fun conv_ToR_add_tag' ctm =
  case Thm.term_of ctm
    of Const(\<^const_name>‹Transformation›, _) $ _ $ _ $ _ =>
        conv_ToR_add_tag ctm
     | _ => let
        fun chk_all_transformation_is_tagged (Const (\<^const_name>‹Action_Tag›, _) $ (Const (\<^const_name>‹Transformation›, _) $ _ $ _ $ _) $ _) = true
          | chk_all_transformation_is_tagged (Const (\<^const_name>‹Transformation›, _) $ _ $ _ $ _) = false
          | chk_all_transformation_is_tagged (X $ Y) = chk_all_transformation_is_tagged X andalso chk_all_transformation_is_tagged Y
          | chk_all_transformation_is_tagged (Abs (_, _, X)) = chk_all_transformation_is_tagged X
          | chk_all_transformation_is_tagged _ = true
     in if chk_all_transformation_is_tagged (Thm.term_of ctm)
        then Conv.all_conv ctm
        else err ctm
    end

fun chk_ToR_prop ctxt term = let
        datatype TP_mode = TP | TP' | CLEAN

        fun err_not_var () = raise TERM ("The source value of a transformation rule must be a variable", [term])
        fun source_val (Const(\<^const_name>‹Pure.imp›, _) $ _ $ X) = source_val X
          | source_val (Const(\<^const_name>‹Pure.all›, _) $ X) = source_val X
          | source_val (Abs (_,_,X)) = source_val X
          | source_val (Const(\<^const_name>‹Action_Tag›, _) $ X $ _) = source_val X
          | source_val (Const(\<^const_name>‹Transformation›, _) $ (Const(\<^const_name>‹φType›, _) $ x $ _) $ _ $ _) =
              (case x of Free _ => ()
                       | Var _  => ()
                       | _ => err_not_var ())
          | source_val (A $ B) = (source_val A ; source_val B)
          | source_val _ = ()
        val _ = if Config.get_generic ctxt chk_source_val then source_val term else ()

        fun set (concl,NONE) X = (concl,SOME X)
          | set (_,SOME _) _ = err' term
        
        fun scan C (Const(\<^const_name>‹Action_Tag›, _) $ X $ Const(\<^const_name>‹𝒯𝒫›, _)) = scan (set C TP) X
          | scan C (Const(\<^const_name>‹Action_Tag›, _) $ X $ Const(\<^const_name>‹𝒯𝒫'›, _)) = scan (set C TP') X
          | scan C (Const(\<^const_name>‹Action_Tag›, _) $ X $ Const(\<^const_name>‹𝒜clean›, _)) = scan (set C CLEAN) X
          | scan _ (Const(\<^const_name>‹Action_Tag›, _) $ _ $ _) = ()
          | scan C (Const(\<^const_name>‹Transformation›, _) $ _ $ _ $ _) = (
              case C
                of (_,SOME TP') => ()
                 | (_,SOME TP) => ()
                 | (_,SOME CLEAN) => ()
                 | (_,NONE) =>
              error "The use of transformation must be tagged in a transformation rule.\n\
                    \You can disable this check by \
                    \‹setup ‹Config.put_global (Phi_Syntax.enable_auto_chk_and_conv) false››.\n\
                    \if you are sure about what you are doing." )
          | scan (concl,C) (Const(\<^const_name>‹Pure.imp›, _) $ A $ B) = (scan (false,C) A; scan (concl,C) B)
          | scan C (X $ Y) = (scan C X ; scan C Y)
          | scan C (Abs (_, _, X)) = scan C X
          | scan _ _ = ()
   in scan (true,NONE) term
  end

fun ToR_rule_add_tag ctxt th =
  Conv.fconv_rule (Phi_Conv.hhf_conv (Phi_Conv.hhf_concl_conv (K (HOLogic.Trueprop_conv conv_ToR_add_tag')))
                                     (K (HOLogic.Trueprop_conv conv_ToR_add_tag')) ctxt) th
  |> (fn th => (chk_ToR_prop (Context.Proof ctxt) (Thm.prop_of th) ; th))
