fun dest_synt_numeral (Const (\<^const_syntax>‹zero_class.zero›, _)) = 0
  | dest_synt_numeral (Const (\<^const_syntax>‹one_class.one›, _)) = 1
  | dest_synt_numeral (Const (\<^const_syntax>‹num.One›, _)) = 1
  | dest_synt_numeral (Const (\<^const_syntax>‹num.Bit0›, _) $ bs) = 2 * dest_synt_numeral bs
  | dest_synt_numeral (Const (\<^const_syntax>‹num.Bit1›, _) $ bs) = 2 * dest_synt_numeral bs + 1
  | dest_synt_numeral (Const (\<^const_syntax>‹numeral_class.numeral›, _) $ X) = dest_synt_numeral X
  | dest_synt_numeral (Const (\<^const_syntax>‹Suc›, _) $ X) = dest_synt_numeral X + 1
  | dest_synt_numeral t = raise TERM ("dest_num", [t])

local

val numT = Type ("Num.num", []);

val one_const = Const ("Num.num.One", numT)
and bit0_const = Const ("Num.num.Bit0", numT --> numT)
and bit1_const = Const ("Num.num.Bit1", numT --> numT)

in
fun mk_numeral_nat 0 = \<^term>‹0::nat›
  | mk_numeral_nat 1 = \<^term>‹1::nat›
  | mk_numeral_nat i =
  let
    fun mk 1 = one_const
      | mk i =
          let
            val (q, r) = Integer.div_mod i 2
          in (if r = 0 then bit0_const else bit1_const) $ mk q end
  in
    if i > 0
    then \<^term>‹numeral::num ⇒ nat› $ mk i 
    else raise TERM ("mk_numeral: " ^ string_of_int i, [])
  end
end
