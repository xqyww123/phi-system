theory PhiTest_Arithmetic
  imports
    Phi_Semantics.PhiSem_CF_Routine
    Phi_Semantics.PhiSem_CF_Breakable
    Phi_Semantics.PhiSem_Variable
    Phi_Semantics.PhiSem_Int_ArbiPrec
    "HOL-Computational_Algebra.Primes"
begin

declare [[φhide_techinicals]]
declare [[ML_print_depth = 1000]]

proc test_prime:
  input ‹𝗏𝖺𝗅 x ⦂ ℕ›
  output ‹𝗏𝖺𝗅 prime x ⦂ 𝔹› ― ‹\<^term>‹prime :: nat => bool› is a predicate checking primes›
  is [routine]
  ❴
    if ‹$x ≤ 1› ❴            (* v this proof is generated by sledgehammer*)
      return (False) affirm using leD prime_gt_1_nat the_φ by auto
    ❵.
    ❴
      ‹2 ⦂ ℕ› → var v ;;

      while ‹i ⦂ 𝗏𝖺𝗋[v] ℕ 𝗌𝗎𝖻𝗃 i.
              Inv: (1 < i ∧ i ≤ x ∧ (∀j ∈ {1<..<i}. ¬ j dvd x)) ∧
              Guard: (i ≠ x) ∧
              End: (i = x)› ― ‹Specification of the loop›
      ❴ ‹$v ≠ $x› ❵. ― ‹Code for loop guard›
      ❴               ― ‹Code for loop body›
        if ‹$x mod $v = 0› ❴               (* v this proof is generated by sledgehammer*)
          return (False) affirm by (metis le_neq_implies_less mod_0_imp_dvd nat_neq_iff prime_nat_not_dvd the_φ(1) the_φ(2) the_φ(4) the_φ(5))
        ❵.
        ❴
          ‹$v + 1› → $v          (* v this proof is generated by sledgehammer*)
          continue affirm by (metis ‹1 < i ∧ i ≤ x ∧ (∀j∈{1<..<i}. ¬ j dvd x)› dvd_imp_mod_0 greaterThanLessThan_iff le_antisym less_Suc_eq not_less_eq_eq the_φ(1) the_φ(2))
        ❵.
      ❵.                       (* v this proof is generated by sledgehammer*)
      return (True) affirm by (metis One_nat_def Primes.prime_nat_def prime_nat_naive the_φlemmata(1) the_φlemmata(3) the_φlemmata(4))
    ❵.
  ❵. .


proc test_prime':
  input ‹𝗏𝖺𝗅 x ⦂ ℕ›
  output ‹𝗏𝖺𝗅 prime x ⦂ 𝔹›
  is [routine]
  ❴
    if ‹$x ≤ 1› ❴
      return (False) affirm using prime_nat_iff the_φ by auto 
    ❵.
    ❴
      ‹2 ⦂ ℕ› → var v ;;
      (* In the previous example, the loop iterates from 2 to x, here we apply an optimization
         where the loop only needs to iterate to sqrt(x). *)
      while ‹i ⦂ 𝗏𝖺𝗋[v] ℕ 𝗌𝗎𝖻𝗃 i.
          Inv: (1 < i ∧ i ≤ x ∧ (∀j ∈ {1<..<i}. ¬ j dvd x)) ∧
          Guard: (i * i ≤ x) ∧
          End: (x < i * i)›
      ❴ ‹$v * $v ≤ $x› ❵.
      ❴
        if ‹$x mod $v = 0› ❴
          return (False) affirm by (metis dvd_mult_cancel2 le_antisym le_neq_implies_less le_square less_not_refl mod_0_imp_dvd prime_gt_0_nat prime_nat_not_dvd the_φ(1) the_φ(2) the_φ(4) the_φ(5))  
        ❵.
        ❴
          ‹$v + 1› → $v
          continue affirm by (metis ‹1 < i ∧ i ≤ x ∧ (∀j∈{1<..<i}. ¬ j dvd x)› dvd_eq_mod_eq_0 gcd_nat.refl greaterThanLessThan_iff leI le_less_Suc_eq less_SucI the_φ(1)) 
        ❵.
      ❵.
      return (True) affirm (*with this optimization, the final obligation fails to be solved
                             automatically, but the manual proof is intuitive and semi-automated.*)
        proof simp
        have ‹False› if assm: ‹¬ prime x›
          proof -
            obtain k where ‹k dvd x ∧ 1 < k ∧ k < x›
              by (metis One_nat_def Suc_lessI assm dvd_pos_nat linorder_le_less_linear nat_dvd_not_less not_less_iff_gr_or_eq prime_nat_iff the_φ(5) zero_less_iff_neq_zero)
            then have ‹k < ib ∨ x div k < ib›
              by (metis dvd_mult_div_cancel leD leI mult_le_mono the_φlemmata(4))
            then show False
              by (metis One_nat_def ‹k dvd x ∧ 1 < k ∧ k < x› div_mod_decomp dvd_imp_mod_0 dvd_triv_right greaterThanLessThan_iff mult.commute nat_arith.rule0 nat_mult_1_right nat_mult_less_cancel_disj the_φlemmata(3))
          qed
        then show ‹prime x›
          by fastforce
        qed
    ❵. ― ‹Close the top branch›
  ❵. ― ‹Close the function body› .



end
