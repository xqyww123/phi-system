section ‹Map Representation of a Tree›

type_synonym ('k,'v) tree = ‹'k list ⇒ 'v›

text ‹This section presents a representation of tree using the mapping from path to value,
    of type \<^typ>‹'key list ⇒ 'val›.
  It implements the hierarchical algebra and supports for permission algebra.

  Basic operations include `push_map` and `pull_map` which put a sub-tree onto certain location
    and fetch a sub-tree at certain location respectively.

  In this representation, the indexes near the root locate at the left side of the list, e.g.,
  ‹[a,b,c] ‣⇩m D› characterizes tree ‹Root → a → b → c → D›.
›

subsection ‹Push a map to a location›

definition push_map :: ‹'a list ⇒ ('a list ⇒ 'b) ⇒ ('a list ⇒ 'b::one)› (infixr "‣⇩m" 75)
  where ‹push_map idx f = (λx. if take (length idx) x = idx then f (drop (length idx) x) else 1 )›

lemma push_map_cons[simp]:
  ‹ push_map (h#depth) f (h#L) = push_map depth f L ›
  unfolding push_map_def
  by clarsimp

lemma push_map_cons_neq:
  ‹ h ≠ h2
⟹ push_map (h#depth) f (h2#L) = 1 ›
  unfolding push_map_def
  by clarsimp

lemma push_map_Nil[simp]:
  ‹ push_map [] f L = f L ›
  unfolding push_map_def
  by clarsimp

lemma push_map_access_Nil[simp]:
  ‹ ofs ≠ []
⟹ push_map ofs f [] = 1 ›
  unfolding push_map_def
  by clarsimp

lemma push_map_access_less:
  ― ‹not added to the default simpset because of heavy computation cost›
  ‹ length L < length ofs
⟹ push_map ofs f L = 1 ›
  unfolding push_map_def
  by clarsimp

lemma push_map_unit[simp]:
  ‹push_map ia (1(ib := x)) = 1(ia@ib := x)›
  unfolding push_map_def
  by (simp add: fun_eq_iff, metis append_eq_conv_conj)

lemma push_map_push_map:
  ‹push_map ia (push_map ib f) = push_map (ia@ib) f›
  unfolding push_map_def fun_eq_iff
  by (smt (verit, ccfv_threshold) add.commute append_eq_append_conv append_eq_conv_conj drop_drop length_append take_add take_drop)

lemma push_map_distrib_mult:
  ‹push_map idx f * push_map idx g = push_map idx (f * g)›
  for f :: ‹'a list ⇒ 'b::monoid_mult›
  unfolding push_map_def fun_eq_iff times_fun_def by simp

lemma push_map_distrib_map_add:
  ‹push_map idx (f ++ g) = push_map idx f ++ push_map idx g›
  unfolding push_map_def fun_eq_iff map_add_def by simp

lemma push_map_sep_disj[simp]:
  ‹push_map idx a ## push_map idx b ⟷ a ## b›
  for a :: ‹'k list ⇒ 'v::sep_magma_1›
  unfolding sep_disj_fun_def push_map_def apply simp
  by (metis append_eq_conv_conj)

lemma push_map_distrib_sep_mult:
  ‹ f ## g
⟹ push_map idx f * push_map idx g = push_map idx (f * g)›
  for f :: ‹'a list ⇒ 'b::sep_magma_1›
  unfolding push_map_def fun_eq_iff times_fun_def by simp

lemma push_map_eq_1[simp]:
  ‹push_map idx f = 1 ⟷ f = 1›
  unfolding push_map_def fun_eq_iff by simp (metis append_eq_conv_conj)

lemma push_map_1[simp]:
  ‹push_map idx 1 = 1›
  unfolding push_map_def fun_eq_iff by simp

lemma push_map_mult_nil[simp]:
  ‹push_map [] f = f›
  unfolding push_map_def fun_eq_iff by simp

lemma share_push_map:
  ‹share n (push_map idx f) = push_map idx (share n f)›
  for f :: ‹'a list ⇒ 'b :: share_one›
  unfolding push_map_def fun_eq_iff share_fun_def by simp

lemma push_map_homo:
  ‹homo_one φ ⟹ φ o (push_map idx f) = push_map idx (φ o f)›
  unfolding push_map_def fun_eq_iff by (simp add: homo_one_def)

lemma push_map_to_share:
  ‹push_map idx (to_share o f) = to_share o (push_map idx f)›
  unfolding push_map_def fun_eq_iff by simp

lemma push_map_dom_eq[simp]:
  ‹dom (push_map idx f) = dom (push_map idx g) ⟷ dom f = dom g›
  unfolding dom_def fun_eq_iff push_map_def set_eq_iff apply simp
  by (metis (full_types) append_eq_conv_conj)

lemma push_map_dom1_eq[simp]:
  ‹dom1 (push_map idx f) = dom1 (push_map idx g) ⟷ dom1 f = dom1 g›
  unfolding dom1_def fun_eq_iff push_map_def set_eq_iff
  by (smt (verit, ccfv_threshold) append_eq_conv_conj mem_Collect_eq)

lemma push_map_mult_ex:
  ‹h ‣⇩m a = b * c ⟹ b ## c ⟹ (∃b' c'. b = h ‣⇩m b' ∧ c = h ‣⇩m c' ∧ a = b' * c' ∧ b' ## c')›
  for a :: ‹'k list ⇒ 'v::sep_no_inverse›
  unfolding fun_eq_iff push_map_def times_fun
  apply auto
  apply (rule exI[where x=‹λk. b (h @ k)›], auto simp add: push_map_def)
  apply (metis append_take_drop_id)
  apply (metis sep_disj_fun sep_no_inverse)
  apply (rule exI[where x=‹λk. c (h @ k)›], auto simp add: push_map_def)
  apply (metis append_take_drop_id)
  apply (metis sep_disj_fun_def sep_no_inverse)
  apply (metis append_eq_conv_conj)
  using sep_disj_fun_def by blast

lemma push_map_mult_same_head[simp]:
  ‹ h ‣⇩m a = h ‣⇩m b * h ‣⇩m c ⟷ a = b * c ›
  for a :: ‹'k list ⇒ 'v::mult_1›
  unfolding fun_eq_iff push_map_def times_fun
  by (auto, metis append_eq_conv_conj)

subsection ‹Pull a map at a location›

definition pull_map :: ‹'a list ⇒ ('a list ⇒ 'b) ⇒ ('a list ⇒ 'b)›
  where ‹pull_map idx f = (λx. f (idx@x))›

lemma pull_map_unit[simp]:
  ‹pull_map ia (1((ia@ib) := x)) = 1(ib := x)›
  unfolding pull_map_def by (simp add: fun_eq_iff)

lemma pull_push_map[simp]:
  ‹pull_map idx (push_map idx f) = f›
  unfolding pull_map_def push_map_def fun_eq_iff by simp

lemma push_pull_map:
  ‹push_map idx (pull_map idx f) ⊆⇩m f›
  unfolding pull_map_def push_map_def map_le_def dom_def
  by simp (metis append_take_drop_id)

lemma pull_map_1[simp]:
  ‹pull_map idx 1 = 1›
  unfolding pull_map_def fun_eq_iff by simp

lemma pull_map_0[simp]:
  ‹pull_map idx 0 = 0›
  unfolding pull_map_def fun_eq_iff by simp

lemma pull_map_nil[simp]:
  ‹pull_map [] f = f›
  unfolding pull_map_def fun_eq_iff by simp

lemma pull_map_pull_map[simp]:
  ‹pull_map a (pull_map b f) = pull_map (b@a) f›
  unfolding pull_map_def fun_eq_iff by simp

lemma pull_map_cons:
  ‹pull_map a (pull_map [b] f) = pull_map (b#a) f›
  unfolding pull_map_def fun_eq_iff by simp

lemma pull_map_funcomp:
  ‹homo_one φ ⟹ φ o (pull_map idx f) = pull_map idx (φ o f)›
  unfolding pull_map_def fun_eq_iff by simp

lemma pull_map_homo_mult:
  ‹pull_map idx (f * g) = pull_map idx f * pull_map idx g›
  unfolding pull_map_def fun_eq_iff
  by (simp add: times_fun)

lemma pull_map_to_share:
  ‹pull_map idx (to_share o f) = to_share o (pull_map idx f)›
  unfolding pull_map_def fun_eq_iff by simp

lemma pull_map_share:
  ‹pull_map idx (share n f) = share n (pull_map idx f)›
  unfolding pull_map_def fun_eq_iff share_fun_def by simp

lemma pull_map_sep_disj[simp]:
  ‹f ## g ⟹ pull_map idx f ## pull_map idx g›
  unfolding pull_map_def sep_disj_fun_def by simp

lemma pull_map_dom_eq:
  ‹dom a = dom b ⟹ dom (pull_map idx a) = dom (pull_map idx b)›
  by (induct idx; simp; simp add: dom_def set_eq_iff pull_map_def)

lemma pull_map_map_add:
  ‹pull_map idx (f ++ g) = pull_map idx f ++ pull_map idx g›
  unfolding pull_map_def
  by (simp add: map_add_def)

definition ‹the_subtree  idx f = push_map idx (pull_map idx f)›
definition ‹trim_subtree idx f = (λx. if take (length idx) x = idx then 1 else f x )›

lemma the_subtree_times_trim_subtree[simp]:
  ‹the_subtree idx f * trim_subtree idx f = f›
  for f :: ‹'a list ⇒ 'b::mult_1›
  unfolding fun_eq_iff trim_subtree_def the_subtree_def pull_map_def push_map_def
  by (clarsimp simp add: times_fun; metis append_take_drop_id)

lemma trim_subtree_subtract:
  ‹ a ## trim_subtree idx f
⟹ (a * trim_subtree idx f = f) ⟷ a = the_subtree idx f›
  for f :: ‹'a list ⇒ 'b::discrete_monoid›
  unfolding fun_eq_iff trim_subtree_def the_subtree_def pull_map_def push_map_def
  apply (auto simp add: times_fun sep_disj_fun_def)
  apply (metis append_take_drop_id)
  apply (metis mult_1_class.mult_1_right)
  by (metis append_eq_conv_conj)

lemma the_subtree_subtract:
  ‹ a ## the_subtree idx f
⟹ (a * the_subtree idx f = f) ⟷ a = trim_subtree idx f›

subsubsection ‹For Trees of Numeric Path›

definition nnode :: ‹(nat list ⇒ 'b) list ⇒ nat list ⇒ 'b::one›
  ― ‹A tree node of children L›
  where ‹nnode L = (λidx. case idx of i#idx' ⇒ (if i < length L then (L!i) idx' else 1) | _ ⇒ 1)›

lemma share_nnode:
  ‹share n (nnode L) = nnode (map (share n) L)›
  for L :: ‹(nat list ⇒ 'a::share_one) list›
  unfolding nnode_def fun_eq_iff by (simp add: share_fun_def split: list.split)

lemma pull_map_node:
  ‹pull_map (i#idx) (nnode L) = (if i < length L then pull_map idx (L!i) else 1)›
  unfolding nnode_def pull_map_def fun_eq_iff by simp
