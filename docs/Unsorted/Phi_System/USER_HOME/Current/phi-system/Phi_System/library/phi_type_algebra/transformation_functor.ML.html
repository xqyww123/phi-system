fun seq_cross seqs =
  let fun cross ret [] = Seq.single (rev ret)
        | cross ret (s :: seqs) =
            Seq.maps (fn x => cross (x::ret) seqs) s
   in cross [] seqs
  end

fun seq_fold_cross seqs ctxt =
  let fun cross (ret, ctxt) [] = Seq.single (rev ret, ctxt)
        | cross (ret, ctxt) (s :: seqs) =
            Seq.maps (fn (x, ctxt') => cross (x::ret, ctxt') seqs) (s ctxt)
   in cross ([], ctxt) seqs 
  end

fun gen_FTF (der_id,name) instruction (phi0 : phi_type) thy =
  let val parse_hint = parse_hint (
            fn Const(\<^const_name>‚ÄπFunctional_Transformation_Functor‚Ä∫, _) $ F1 $ _ $ _ $ _ $ _ $ _ $ _ $ _ => F1
             | Const(\<^const_name>‚ÄπFunctional_Transformation_BiFunctor‚Ä∫, _) $ F1 $ _ $ _ $ _ $ _ $ _ $ _ $ _ $ _ $ _ $ _ $ _ => F1
             | Const(\<^const_name>‚ÄπFunctional_Transformation_Functor‚á©Œõ‚Ä∫, _) $ F1 $ _ $ _ $ _ $ _ $ _ $ _ $ _ => F1
          )
      fun extract_oprs_from_hint (Const(\<^const_name>‚ÄπFunctional_Transformation_Functor‚Ä∫, _) $ _ $ _ $ _ $ _ $ D $ R $ f $ p) = [D,R,f,p]
        | extract_oprs_from_hint (Const(\<^const_name>‚ÄπFunctional_Transformation_BiFunctor‚Ä∫, _) $ _ $ _ $ _ $ _ $ _ $ _ $ D1 $ D2 $ R1 $ R2 $ p $ f)
            = [D1,D2,R1,R2,p,f]
        | extract_oprs_from_hint (Const(\<^const_name>‚ÄπFunctional_Transformation_Functor‚á©Œõ‚Ä∫, _) $ _ $ _ $ _ $ _ $ D $ R $ f $ p) = [D,R,f,p]
        | extract_oprs_from_hint _ = []

     val arity = case #1 instruction of SOME (H, _) =>
                        (case Term.head_of (HOLogic.dest_Trueprop (Logic.strip_imp_concl H))
                           of Const(\<^const_name>‚ÄπFunctional_Transformation_Functor‚Ä∫, _) => 1
                            | Const(\<^const_name>‚ÄπFunctional_Transformation_BiFunctor‚Ä∫, _) => 2
                            | Const(\<^const_name>‚ÄπFunctional_Transformation_Functor‚á©Œõ‚Ä∫, _) => ~1
                            | _ => unsupport ())
                     | _ => length (filter (Phi_Syntax.is_non_parameterized_phityp o snd) (parameters_of phi0))
     val rule = case arity of 1 => @{thm' œÜTA_FTF_rule}
                            | 2 => @{thm' œÜTA_biFTF_rule}
                            | ~1 => @{thm' œÜTA_FTF‚á©Œõ_rule}
                            | _ => unsupport ()

     fun reasoning phi ctxt = exhaustive_PLPR_reasoner NONE (K (chk_unfolded [] ctxt phi)) ctxt
     val simp = if is_some (#1 instruction) then K I
                else simplifier_by_cong (case arity of 1 => @{thms' œÜTA_FTF_deriver_cong}

                                                     | ~1 => @{thms' œÜTA_FTF‚á©Œõ_deriver_cong})

      val rules = deriving_framework (name, parse_hint, extract_oprs_from_hint,

                                    phi0 instruction thy
   in Seq.pull rules
   |> (fn NONE => raise Automation_Fail (fn () => [Pretty.str "Fail to derive any property"])
        | SOME (rule, _) =>
          let (*fun set_prem_to_True rule =

              val rule' = set_prem_to_True rule'0
              
              fun setup_functional_TF cmd lthy =
                let val ((_,[locale_rule']), _) = Variable.import true [rule'] lthy
                    val interp_name = suffix_name_by_index der_id (Term.term_name (#term phi0))
                 in case Thm.concl_of locale_rule'
                      of \<^Const>‚ÄπTrueprop‚Ä∫ $ (Const(\<^const_name>‚ÄπFunctional_Transformation_Functor‚Ä∫, _)
                            $ Fa $ Fb $ D $ R $ m $ Pr $ pm $ fm) =>
                          cmd ([(\<^locale>‚ÄπFunctional_Transformation_Functor‚Ä∫,
