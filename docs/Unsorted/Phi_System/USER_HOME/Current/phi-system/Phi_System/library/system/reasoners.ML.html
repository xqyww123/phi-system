signature PHI_SYS_REASONERS = sig

val is_user_dependent_antecedent : term -> bool
val is_proof_obligation : term -> bool

end

structure Phi_Sys_Reasoner : PHI_SYS_REASONERS = struct

fun is_user_dependent_antecedent (Const(\<^const_name>‹Trueprop›, _) $ X) = is_user_dependent_antecedent X
  | is_user_dependent_antecedent (Const(\<^const_name>‹Argument›, _) $ _ ) = true
  | is_user_dependent_antecedent (Const(\<^const_name>‹Do›, _) $ _ ) = true
  | is_user_dependent_antecedent (Const(\<^const_name>‹ParamTag›, _) $ _ ) = true
  | is_user_dependent_antecedent (Const(\<^const_name>‹φProcedure›, _) $ _ $ _ $ _ $ _) = true
  | is_user_dependent_antecedent (Const(\<^const_name>‹Pure.all›, _) $ Abs (_, _, X)) = is_user_dependent_antecedent X
  | is_user_dependent_antecedent (\<^Const>‹Pure.all ty› $ X) =
      is_user_dependent_antecedent (\<^Const>‹Pure.all ty› $ Abs ("_", ty, Term.incr_boundvars 1 X $ Bound 0))
  | is_user_dependent_antecedent (Const(\<^const_name>‹Pure.imp›, _) $ _ $ X) = is_user_dependent_antecedent X
  | is_user_dependent_antecedent (Const(\<^const_name>‹HOL.implies›, _) $ _ $ X) = is_user_dependent_antecedent X
  | is_user_dependent_antecedent (Const(\<^const_name>‹HOL.All›, _) $ Abs (_, _, X)) = is_user_dependent_antecedent X
  | is_user_dependent_antecedent (\<^Const>‹HOL.All ty› $ X) =
      is_user_dependent_antecedent (\<^Const>‹HOL.All ty› $ Abs ("_", ty, Term.incr_boundvars 1 X $ Bound 0))
  | is_user_dependent_antecedent (Const(\<^const_name>‹Labelled›, _) $ _ $ X) = is_user_dependent_antecedent X
  | is_user_dependent_antecedent (Const(\<^const_name>‹Technical›, _) $ X) = is_user_dependent_antecedent X
(*  | is_user_dependent_antecedent (Const(\<^const_name>‹meta_Ball›, _) $ _ $ Abs (_, _, X)) = is_user_dependent_antecedent X *)
  | is_user_dependent_antecedent _ = false

fun is_proof_obligation (Const(\<^const_name>‹Trueprop›, _) $ X) = is_proof_obligation X
  | is_proof_obligation (Const(\<^const_name>‹Premise›, _) $ Const(\<^const_name>‹default›, _) $ _) = true
  | is_proof_obligation (Const(\<^const_name>‹Premise›, _) $ Const(\<^const_name>‹MODE_COLLECT›, _) $ _) = true
  | is_proof_obligation _ = false

end
