Theory Phi_BI.LCRO_Interval

theory LCRO_Interval
  imports Algebras
begin

section ‹Left Closed Right Open Intervals›

subsection ‹Definition›

typedef (overloaded) 'a lcro_intvl =
  "{(a::'a::preorder, b). a ≤ b}"
  morphisms bounds_of_interval Interval'
  by auto

setup_lifting type_definition_lcro_intvl

definition Interval :: ‹'a::preorder ⇒ 'a ⇒ 'a lcro_intvl› ("[_,_')" [51,51] 1000)
  where ‹[l,u) ≡ Interval' (l,u)›

lift_definition lower::"('a::preorder) lcro_intvl ⇒ 'a" is fst .

lift_definition upper::"('a::preorder) lcro_intvl ⇒ 'a" is snd .

lemma lower_interval[simp]:
  ‹a ≤ b ⟹ lower [a,b) = a›
  unfolding Interval_def lower_def
  by (simp add: Interval'_inverse)

lemma upper_interval[simp]:
  ‹a ≤ b ⟹ upper [a,b) = b›
  unfolding Interval_def upper_def
  by (simp add: Interval'_inverse)

lemma interval_eq_iff: "a = b ⟷ lower a = lower b ∧ upper a = upper b"
  by transfer auto

lemma interval_eqI: "lower a = lower b ⟹ upper a = upper b ⟹ a = b"
  by (auto simp: interval_eq_iff)

lemma lower_le_upper[simp]: "lower i ≤ upper i"
  by transfer auto


subsubsection ‹Set of›

lift_definition set_of :: "'a::preorder lcro_intvl ⇒ 'a set" is "λx. {fst x ..< snd x}" .
  
lemma set_of_eq: "set_of x = {lower x ..< upper x}"
  by transfer simp


subsection ‹Properties›

subsubsection ‹Equal›

instantiation lcro_intvl :: ("{preorder,equal}") equal
begin

definition "equal_class.equal a b ≡ (lower a = lower b) ∧ (upper a = upper b)"

instance proof qed (simp add: equal_lcro_intvl_def interval_eq_iff)
end


subsubsection ‹Order \& Lattice›

instantiation lcro_intvl :: ("preorder") ord begin

definition less_eq_lcro_intvl :: "'a lcro_intvl ⇒ 'a lcro_intvl ⇒ bool"
  where "less_eq_lcro_intvl a b ⟷ lower b ≤ lower a ∧ upper a ≤ upper b"

definition less_lcro_intvl :: "'a lcro_intvl ⇒ 'a lcro_intvl ⇒ bool"
  where  "less_lcro_intvl x y = (x ≤ y ∧ ¬ y ≤ x)"

instance proof qed
end


instantiation lcro_intvl :: ("lattice") semilattice_sup
begin

lift_definition sup_lcro_intvl :: "'a lcro_intvl ⇒ 'a lcro_intvl ⇒ 'a lcro_intvl"
  is "λ(a, b) (c, d). (inf a c, sup b d)"
  by (auto simp: le_infI1 le_supI1)

lemma lower_sup[simp]: "lower (sup A B) = inf (lower A) (lower B)"
  by transfer auto

lemma upper_sup[simp]: "upper (sup A B) = sup (upper A) (upper B)"
  by transfer auto

instance proof qed (auto simp: less_eq_lcro_intvl_def less_lcro_intvl_def interval_eq_iff)
end

lemma set_of_interval_union:
  "set_of A ∪ set_of B ⊆ set_of (sup A B)"
  for A::"'a::lattice lcro_intvl"
  apply (auto simp: set_of_eq le_infI1 less_supI1 )
  using le_infI2 apply blast
  by (simp add: less_supI2)

lemma interval_union_commute: "sup A B = sup B A" for A::"'a::lattice lcro_intvl"
  by (auto simp add: interval_eq_iff inf.commute sup.commute)

lemma interval_union_mono1: "set_of a ⊆ set_of (sup a A)" for A :: "'a::lattice lcro_intvl"
  using set_of_interval_union by blast

lemma interval_union_mono2: "set_of A ⊆ set_of (sup a A)" for A :: "'a::lattice lcro_intvl"
  using set_of_interval_union by blast



subsubsection ‹Partial Addition›

text ‹Defined as concatenating two adjacent intervals.›

instantiation lcro_intvl :: (preorder) partial_add_cancel begin

lift_definition plus_lcro_intvl :: ‹'a lcro_intvl ⇒ 'a lcro_intvl ⇒ 'a lcro_intvl›
  is ‹λ(a,b) (c,d). if b = c then (a,d) else (undefined, undefined)›
  by (clarsimp; metis Pair_inject dual_order.trans order_eq_refl)

lift_definition dom_of_add_lcro_intvl :: ‹'a lcro_intvl ⇒ 'a lcro_intvl ⇒ bool›
  is ‹λ(a,b) (c,d). b = c› .

lemma dom_of_add_lcro_intvl[simp]:
  ‹a ##⇩+ b ⟷ upper a = lower b›
  by (transfer; clarsimp)

lemma add_lcro_intvl_inj[simp]:
  ‹a ##⇩+ b ⟹ lower (a + b) = lower a›
  ‹a ##⇩+ b ⟹ upper (a + b) = upper b›
  by (transfer; clarsimp)+

instance by (standard; clarsimp simp add: interval_eq_iff)
end

lemma dom_of_add_lcro_intvl'[simp]:
  ‹⟦ a ≤ b ; c ≤ d ⟧ ⟹ [a,b) ##⇩+ [c,d) ⟷ b = c›
  by simp

lemma add_lcro_intvl [simp]:
  ‹⟦ a ≤ b ; b ≤ c ⟧ ⟹ [a,b) + [b,c) = [a,c)›
  using interval_eq_iff order_trans by fastforce

lemma set_of_add_interval:
  ‹a ##⇩+ b ⟹ set_of (a + b) = set_of a ∪ set_of b›
  for a :: ‹'a::linorder lcro_intvl›
  by (clarsimp simp add: set_of_eq; metis ivl_disj_un_two(3) lower_le_upper)

lemma additive_join_sub_interval:
  ‹y ≼⇩+ z ⟷ upper y = upper z ∧ lower z ≤ lower y›
  unfolding additive_join_sub_def
  by (simp add: interval_eq_iff, rule,
      metis add_lcro_intvl_inj dom_of_add_lcro_intvl lower_le_upper,
      rule exI[where x=‹[lower z, lower y)›]; clarsimp)

instance lcro_intvl :: (order) positive_partial_add_magma
  by (standard;
      clarsimp simp add: additive_join_sub_def interval_eq_iff;
      metis lower_le_upper order_antisym)

text ‹lcro_intvl is not strict positive as there are zero element [x,x)›

instance lcro_intvl :: (order) partial_semigroup_add
  by (standard; clarsimp simp add: interval_eq_iff)

text ‹lcro_intvl is also a good example where it is not dom_of_add_intuitive,
      and also where has unique neutral for each element but not a unique one.›

instance lcro_intvl :: (order) partial_cancel_semigroup_add
  by (standard; clarsimp simp add: interval_eq_iff)


subsubsection ‹Membership›

definition interval_membership :: ‹'a::preorder ⇒ 'a lcro_intvl ⇒ bool› (infix "∈⇩I" 50)
  where [simp]: ‹x ∈⇩I i ⟷ lower i ≤ x ∧ x < upper i›

lemma not_in_empty_interval[simp]:
  ‹¬ (x ∈⇩I [a,a))›
  for x :: ‹'a::order›
  by (simp add: leD)

subsubsection ‹Width›

abbreviation width_of :: ‹'a::{preorder, minus} lcro_intvl ⇒ 'a›
  where ‹width_of i ≡ (upper i - lower i)›

hide_const (open) width_of lower upper
