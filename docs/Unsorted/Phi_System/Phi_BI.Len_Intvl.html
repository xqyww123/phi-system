Theory Phi_BI.Len_Intvl

theory Len_Intvl
  imports Algebras
begin

section ‹Left Closed Right Open Intervals in the Length representation›

subsection ‹Definition›

datatype 'a len_intvl = Len_Intvl (start: 'a) (len: nat)

notation Len_Intvl ("⟦_ : _⦆")

lemma len_intvl_all: ‹All P ⟷ (∀s t. P ⟦s:t⦆)›
  by (metis len_intvl.collapse)

lemma len_intvl_ex: ‹Ex P ⟷ (∃s t. P ⟦s : t⦆)›
  by (metis len_intvl.collapse)

lemma ‹Pure.all P ≡ (⋀s t. PROP P ⟦s : t⦆)›
proof
  fix s t
  assume ‹⋀x. PROP P x›
  then show ‹PROP P ⟦s : t⦆› .
next
  fix x
  assume ‹⋀s t. PROP P ⟦s : t⦆›
  from ‹PROP P ⟦start x : len x⦆›
  show ‹PROP P x› by simp
qed


subsubsection ‹Helper›

class shift_by_nat =
  fixes shift_by_nat :: ‹'a ⇒ nat ⇒ 'a›
  assumes shift_by_nat_assoc: ‹shift_by_nat (shift_by_nat x n) m = shift_by_nat x (n + m)›

class shift_by_nat_ord = shift_by_nat + linorder +
  assumes shift_by_nat_ord: ‹x ≤ shift_by_nat x n›

subsection ‹Operations›

subsubsection ‹Set›

definition set :: ‹'a::shift_by_nat_ord len_intvl ⇒ 'a set›
  where [iff]: ‹set i = {start i ..< shift_by_nat (start i) (len i)}›

lemma forall[simp]:
  ‹ (∀i∈{len_intvl.start iv..<len_intvl.start iv + len_intvl.len iv}. P i) ⟷
    (∀i < len_intvl.len iv. P (i + len_intvl.start iv)) ›
  by (auto; metis add.commute le_Suc_ex nat_add_left_cancel_less)

lemma forall_alt[simp]:
  ‹ (∀i. len_intvl.start iv ≤ i ∧ i < len_intvl.start iv + len_intvl.len iv ⟶ P i) ⟷
    (∀i < len_intvl.len iv. P (i + len_intvl.start iv)) ›
  by (auto; metis add.commute le_add_diff_inverse less_diff_conv2)

subsection ‹List›

abbreviation list⇩N :: ‹nat len_intvl ⇒ nat list›
  where ‹list⇩N iv ≡ [start iv ..< start iv + len iv]›


subsection ‹Properties›

subsubsection ‹Partial Addition› ― ‹as concatenating two adjacent intervals›

instantiation len_intvl :: (shift_by_nat) partial_add_cancel begin

definition plus_len_intvl :: ‹'a len_intvl ⇒ 'a len_intvl ⇒ 'a len_intvl›
  where ‹plus_len_intvl x y = ⟦ start x : len x + len y ⦆›

definition dom_of_add_len_intvl :: ‹'a len_intvl ⇒ 'a len_intvl ⇒ bool›
  where ‹dom_of_add_len_intvl x y ⟷ shift_by_nat (start x) (len x) = start y›

lemma plus_len_intvl[iff]:
  ‹ ⟦ a : b ⦆ + ⟦ c : d ⦆ = ⟦ a : b + d ⦆  ›
  unfolding plus_len_intvl_def
  by simp

lemma dom_of_add_len_intvl[iff]:
  ‹ a ##⇩+ b ⟷ shift_by_nat (start a) (len a) = (start b) ›
  unfolding dom_of_add_len_intvl_def
  by (cases a; cases b; simp)

lemma plus_len_intvl_start[simp]:
  ‹ start (a + b) = start a ›
  by (cases a; cases b; simp)

lemma plus_len_intvl_len[simp]:
  ‹ len (a + b) = len a + len b ›
  by (cases a; cases b; simp)

instance by (standard; case_tac a; case_tac b; case_tac c; simp)

end

lemma set_add:
  ‹x ##⇩+ y ⟹ set x ∪ set y = set (x + y) ›
  by (cases x; cases y;
      simp add: shift_by_nat_assoc[symmetric];
      meson ivl_disj_un_two(3) shift_by_nat_ord)

subsection ‹Instances›

instantiation nat :: shift_by_nat_ord begin
definition shift_by_nat_nat :: ‹nat ⇒ nat ⇒ nat›
  where [iff]: ‹shift_by_nat_nat ≡ (+)›
instance by (standard; simp)
end

instantiation int :: shift_by_nat_ord begin
definition shift_by_nat_int :: ‹int ⇒ nat ⇒ int›
  where [iff]: ‹shift_by_nat_int x y ≡ x + of_nat y›
instance by (standard; simp)
end


hide_const (open) start len set
hide_fact (open) set_add forall forall_alt

end
