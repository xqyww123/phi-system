Theory Dynamic_Array

theory Dynamic_Array
  imports Phi_Semantics.PhiSem_C
          Phi_Semantics.PhiSem_Mem_C_MI
          PhiStd.PhiStd_Slice
begin

declare [[φtrace_reasoning = 1]]

φtype_def DynArr
  where ‹l ⦂ DynArr addr T ≡ (a⇩D, len, cap) ⦂ 𝗈𝖻𝗃[addr] ⦃ data: Ptr[𝖺𝗋𝗋𝖺𝗒[cap] (𝗍𝗒𝗉𝖾𝗈𝖿 T)], len: ℕ(𝗌𝗂𝗓𝖾_𝗍), cap: ℕ(𝗌𝗂𝗓𝖾_𝗍) ⦄,
                             data ⦂ 𝗈𝖻𝗃[a⇩D] 𝔸𝗋𝗋𝖺𝗒[cap] T
         𝗌𝗎𝖻𝗃 a⇩D len cap data. len = length l ∧ cap = length data ∧
                              len ≤ cap ∧ (cap = 0 ∨ cap < 2 * len) ∧
                              take len data = l ∧
                              𝗍𝗒𝗉𝖾𝗈𝖿 T ≠ 𝗉𝗈𝗂𝗌𝗈𝗇 ›

  deriving ‹Abstract_Domain T P ⟹ Abstract_Domain (DynArr addr T) (λl. list_all P l ∧ addr ≠ 0 ∧ 𝗍𝗒𝗉𝖾𝗈𝖿 T ≠ 𝗉𝗈𝗂𝗌𝗈𝗇)›
       and ‹Object_Equiv T eq ⟹ Object_Equiv (DynArr addr T) (list_all2 eq)›
       and ‹ 𝖼𝗈𝗇𝖽𝗂𝗍𝗂𝗈𝗇 (𝗍𝗒𝗉𝖾𝗈𝖿 T = 𝗍𝗒𝗉𝖾𝗈𝖿 U ∧ addr' = addr)
         ⟹ Transformation_Functor (DynArr addr) (DynArr addr') T U (λ_. UNIV) (λ_. UNIV) list_all2›
       and Pointer_Of


abbreviation ‹𝖽𝗒𝗇𝖺𝗋𝗋 ≡ 𝗌𝗍𝗋𝗎𝖼𝗍 {data: 𝗉𝗍𝗋, len: 𝗌𝗂𝗓𝖾_𝗍, cap: 𝗌𝗂𝗓𝖾_𝗍}›



proc len_dynarr:
  input    ‹l ⦂ 𝗋𝖾𝖿 DynArr addr T›
  output   ‹length l ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟ l ⦂ DynArr addr T›
  unfolding DynArr.unfold
❴
  addr.len
❵ .


proc get_dynarr:
  input    ‹l ⦂ 𝗋𝖾𝖿 DynArr addr T❟ i ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)›
  premises ‹i < length l›
  output   ‹l!i ⦂ 𝗏𝖺𝗅 T❟ l ⦂ DynArr addr T›
  unfolding DynArr.unfold
❴
  addr.data[i]
❵ .


proc set_dynarr:
  input    ‹l ⦂ 𝗋𝖾𝖿 DynArr addr T❟ i ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟ v ⦂ 𝗏𝖺𝗅 T›
  premises ‹i < length l›
  output   ‹l[i := v] ⦂ DynArr addr T›
  unfolding DynArr.unfold
❴
  addr.data[i] := v
❵ .

proc Max:
  input  ‹x ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟ y ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)›
  output ‹max x y ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)›
❴
  if (x < y) ❴ y ❵ ❴ x ❵
❵ .


proc push_dynarr:
  input    ‹l ⦂ 𝗋𝖾𝖿 DynArr addr T❟ v ⦂ 𝗏𝖺𝗅 T›
  premises ‹length l ≤ 2^(addrspace_bits-2) ∧ 2 ≤ addrspace_bits›
  output   ‹l + [v] ⦂ DynArr addr T›
❴
  𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 𝗈𝗉𝖾𝗇 ；
  val len ← addr.len ；
  val cap ← addr.cap ；
  if (cap = len) ❴
      val cap' ← Max(cap * 2, 1) ；
      val data' ← calloc (cap') ‹T› ；
      memcpy (data', addr.data , len) ；
      mfree (addr.data) ；
      addr.data := data' ；
      addr.len := addr.len + 1 ；
      addr.cap := cap' ；
      data'[len] := v ；
      𝗆𝖺𝗄𝖾𝗌 ‹l + [v] ⦂ DynArr addr _›
  ❵ ❴
      addr.data[len] := v ；
      addr.len := len + 1 ；
      𝗆𝖺𝗄𝖾𝗌 ‹l + [v] ⦂ DynArr addr _›
  ❵
❵ .

proc concat_dynarr:
  input   ‹l1 ⦂ 𝗋𝖾𝖿 DynArr addr1 T❟ l2 ⦂ 𝗋𝖾𝖿 DynArr addr2 T›
  premises ‹length l1 + length l2 < 2^(addrspace_bits-2) ∧ 2 ≤ addrspace_bits›
  requires ‹Semantic_Zero_Val (𝗍𝗒𝗉𝖾𝗈𝖿 T) T zero›
  output  ‹l1 + l2 ⦂ DynArr addr1 T❟ l2 ⦂ DynArr addr2 T›
❴
  val len ← len_dynarr (addr2) ；

  iterate (0, len) ‹λi. l1 + take i l2 ⦂ DynArr addr1 T›
  ❴ → val i ；
    push_dynarr (addr1, get_dynarr (addr2, i))
  ❵
❵ .

proc pop_dynarr:
  input    ‹l ⦂ 𝗋𝖾𝖿 DynArr addr T❟ v ⦂ 𝗏𝖺𝗅 T›
  premises ‹l ≠ [] ∧ 2 ≤ addrspace_bits›
  requires ‹Semantic_Zero_Val (𝗍𝗒𝗉𝖾𝗈𝖿 T) T zero›
  output   ‹last l ⦂ 𝗏𝖺𝗅 T❟ butlast l ⦂ DynArr addr T›
❴
  𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 𝗈𝗉𝖾𝗇 ；
  val len ← addr.len - 1 ；
  val half_cap ← addr.cap / 2 ；
  val ret ← addr.data[len] ；
  addr.len := len ；
  if (len ≤ half_cap) ❴
    holds_fact [simp]: ‹length l - Suc 0 = length ya div 2› ；
    val data' ← calloc (half_cap) ‹T› ；
    memcpy (data', addr.data, len) ；
    mfree (addr.data) ；
    addr.data := data' ；
    addr.cap := half_cap ； 
    𝗆𝖺𝗄𝖾𝗌 ‹DynArr addr _›
  ❵ ❴ 
    𝗆𝖺𝗄𝖾𝗌 ‹DynArr addr _›
  ❵
  ret
❵ .


proc new_dynarr:
  input  ‹Void›
  premises ‹𝗍𝗒𝗉𝖾𝗈𝖿 T ≠ 𝗉𝗈𝗂𝗌𝗈𝗇›
  requires ‹Semantic_Zero_Val (𝗍𝗒𝗉𝖾𝗈𝖿 T) T zero›
  output ‹[] ⦂ 𝗋𝖾𝖿 DynArr addr T 𝗌𝗎𝖻𝗃 addr. ⊤›
❴
  val ret ← calloc1 ‹⦃ data: Ptr[𝖺𝗋𝗋𝖺𝗒[0] (𝗍𝗒𝗉𝖾𝗈𝖿 T)], len: ℕ(𝗌𝗂𝗓𝖾_𝗍), cap: ℕ(𝗌𝗂𝗓𝖾_𝗍) ⦄› ；
  ret.data := (calloc (‹0 ⦂ ℕ(𝗌𝗂𝗓𝖾_𝗍)›) ‹T›) ；
  𝗆𝖺𝗄𝖾𝗌 ‹DynArr addr _› ；
  ret
❵ .


proc del_dynarr:
  input  ‹l ⦂ 𝗋𝖾𝖿 DynArr addr T›
  output ‹Void›
❴
  𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 𝗈𝗉𝖾𝗇 ；
  mfree (addr.data) ；
  mfree (addr)
❵ .

proc map_dynarr:
  input  ‹l ⦂ 𝗋𝖾𝖿 DynArr addr T›
  requires C: ‹⋀x u. 𝗉𝗋𝗈𝖼 C u ⦃ x ⦂ 𝗏𝖺𝗅[u] T ⟼ f x ⦂ 𝗏𝖺𝗅 T ⦄ ›
  output ‹map f l ⦂ DynArr addr T›
❴
  note [φsledgehammer_simps] = list_eq_iff_nth_eq nth_append ；
  iterate (‹0 ⦂ ℕ(𝗌𝗂𝗓𝖾_𝗍)›, len_dynarr (addr)) ‹λi. (map f (take i l) @ drop i l) ⦂ DynArr addr T›
  ❴ → val i ；
     set_dynarr (addr, i, C (get_dynarr (addr, i)))
  ❵ ；
❵ .

proc exists_dynarr:
  input  ‹l ⦂ 𝗋𝖾𝖿 DynArr addr T›
  requires C: ‹⋀x u. 𝗉𝗋𝗈𝖼 C u ⦃ x ⦂ 𝗏𝖺𝗅[u] T ⟼ P x ⦂ 𝗏𝖺𝗅 𝔹 ⦄ ›
  output ‹list_ex P l ⦂ 𝗏𝖺𝗅 𝔹❟ l ⦂ DynArr addr T›
❴
  var zz ← False ；
  iterate (‹0 ⦂ ℕ(𝗌𝗂𝗓𝖾_𝗍)›, len_dynarr (addr)) ‹λi. l ⦂ DynArr addr T❟ list_ex P (take i l) ⦂ 𝗏𝖺𝗋[zz] 𝔹›
    ❴ → val i ；
      zz ∨ C (get_dynarr (addr, i)) → zz
    ❵ ；
  zz
❵ .


proc fold_map_dynarr:
  input  ‹l ⦂ 𝗋𝖾𝖿 DynArr addr T❟ z0 ⦂ 𝗏𝖺𝗅 U›
  premises ‹𝗍𝗒𝗉𝖾𝗈𝖿 U ≠ 𝗉𝗈𝗂𝗌𝗈𝗇›
  requires C: ‹⋀x z u v. 𝗉𝗋𝗈𝖼 C u v ⦃ x ⦂ 𝗏𝖺𝗅[u] T❟ z ⦂ 𝗏𝖺𝗅[v] U ⟼ f x ⦂ 𝗏𝖺𝗅 T❟ g x z ⦂ 𝗏𝖺𝗅 U ⦄ ›
  output ‹fold g l z0 ⦂ 𝗏𝖺𝗅 U❟ map f l ⦂ DynArr addr T›
❴
  var zz ← z0 ；
  iterate (‹0 ⦂ ℕ(𝗌𝗂𝗓𝖾_𝗍)›, len_dynarr (addr))
           ‹λi. fold g (take i l) z0 ⦂ 𝗏𝖺𝗋[zz] U❟ (map f (take i l) @ drop i l) ⦂ DynArr addr T›
  ❴ → val i ；
    C (get_dynarr (addr, i), zz) → val x', var zz ;;
    set_dynarr (addr, i, x')
  ❵ certified by (auto simp add: list_eq_iff_nth_eq nth_append, auto_sledgehammer,
                  insert ‹i < length l›, induct i, auto simp add: take_Suc_conv_app_nth) ;;
  zz
❵ .


text ‹The Conclusions of above Certification is the following Specification Theorems›
