Theory Phi_Semantics.PhiSem_Mem_C_MI

theory PhiSem_Mem_C_MI ― ‹MI: Machine Integer›
  imports PhiSem_Mem_C PhiSem_Machine_Integer PhiSem_Aggregate_Array
begin


debt_axiomatization
      MemObj_Size_int: ‹MemObj_Size (sem_int_T n) = 0 ⟷ n = 0›

lemma phantom_mem_semantic_type_𝖺𝗂𝗇𝗍[simp]:
  ‹ phantom_mem_semantic_type (sem_int_T n) ⟷ n = 0 ›
  unfolding phantom_mem_semantic_type_def
  using MemObj_Size_int by clarsimp

abbreviation ‹𝗌𝗂𝗓𝖾_𝗍 ≡ 𝗂𝗇𝗍(𝗌𝗂𝗓𝖾_𝗍)›
 

proc calloc:
  requires ‹𝗉𝖺𝗋𝖺𝗆 T›
  input ‹n ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)›
  requires ‹Semantic_Zero_Val TY T z›
  premises ‹TY ≠ 𝗉𝗈𝗂𝗌𝗈𝗇›
  output ‹replicate n z ⦂ 𝗆𝖾𝗆[addr] (𝗆𝖾𝗆-𝖼𝗈𝖾𝗋𝖼𝖾 (Array n T))❟ addr ⦂ 𝗏𝖺𝗅 TypedPtr (𝖺𝗋𝗋𝖺𝗒[n] TY)
          𝗌𝗎𝖻𝗃 addr. addr.offset addr = 0›
  including Semantic_Zero_Val_EIF_brute
❴
  note list_all2_conv_all_nth[simp] ；

  semantic_local_value($n) ‹𝗂𝗇𝗍(𝗌𝗂𝗓𝖾_𝗍)›
  semantic_assert ‹Zero TY ≠ None›
  apply_rule FIC.aggregate_mem.allocate_rule[where TY=‹𝖺𝗋𝗋𝖺𝗒[snd (sem_dest_int (φarg.dest 𝖺𝗋𝗀1))] TY›
                                               and U=‹{sem_mk_array (replicate (snd (sem_dest_int (φarg.dest 𝖺𝗋𝗀1))) (the (Zero TY)))}›]

  semantic_assumption ‹type_storable_in_mem (𝖺𝗋𝗋𝖺𝗒[snd (sem_dest_int (φarg.dest 𝖺𝗋𝗀1))] TY)›

  ‹replicate n z ⦂ MAKE _ (𝗆𝖾𝗆-𝖻𝗅𝗄[blk] (MAKE _ (𝗆𝖾𝗆-𝖼𝗈𝖾𝗋𝖼𝖾 (Array n T))))›
  ‹replicate n z ⦂ MAKE _ (𝗆𝖾𝗆[Addr blk 0] Array n T)›

  have t1: ‹valid_memaddr (Addr blk [])›
    unfolding valid_memaddr_def Valid_MemBlk_def
    using ‹block.layout blk = 𝖺𝗋𝗋𝖺𝗒[n] TY›
    by (cases blk; clarsimp simp: ‹type_storable_in_mem (𝖺𝗋𝗋𝖺𝗒[n] TY)› address_type_def; auto_sledgehammer)
  note address_type_def [φsledgehammer_simps] ；
  
  semantic_return ‹sem_mk_pointer (Addr (φarg.dest 𝗏2) 0) ⊨ (Addr blk 0 ⦂ TypedPtr (𝖺𝗋𝗋𝖺𝗒[n] TY))›
❵ .


proc op_shift_pointer [φoverload +]:
  requires ‹𝗉𝖺𝗋𝖺𝗆 TY›
  input  ‹addr ⦂ 𝗏𝖺𝗅 RawPointer❟ n ⦂ 𝗏𝖺𝗅 ℕ('b::len)›
  output ‹addr ||+ of_nat (MemObj_Size TY * n) ⦂ 𝗏𝖺𝗅 RawPointer›
❴
  $addr semantic_local_value 𝗉𝗍𝗋
  semantic_return ‹
    sem_mk_pointer (sem_dest_pointer (φarg.dest 𝖺𝗋𝗀1) ||+ of_nat (MemObj_Size TY * n))
        ⊨ (addr ||+ of_nat (MemObj_Size TY * n) ⦂ RawPointer)›
❵ .

proc abst_shift_pointer [φoverload +]:
  requires [unfolded abstract_address_offset_def, useful]: ‹abstract_address_offset addr TY TY' n addr'›
  input  ‹addr ⦂ 𝗏𝖺𝗅 TypedPtr TY❟ n ⦂ 𝗏𝖺𝗅 ℕ('b::len)›
  premises ‹addr ≠ 0›
  output ‹addr' ⦂ 𝗏𝖺𝗅 TypedPtr TY'›
❴
  op_shift_pointer ($addr to RawPointer, $n) ‹TY› to ‹TypedPtr TY'›
❵ .
  


end
