Theory Rational_Arith

theory Rational_Arith
  imports Phi_Semantics.PhiSem_C
          Phi_Semantics.PhiSem_Int_ArbiPrec
begin

abbreviation ‹𝗋𝖺𝗍𝗂𝗈𝗇𝖺𝗅 ≡ 𝗌𝗍𝗋𝗎𝖼𝗍{num: 𝖺𝗂𝗇𝗍, den: 𝖺𝗂𝗇𝗍}›

  
φtype_def φRational :: ‹(VAL, rat) φ› ("ℚ")
  where ‹x ⦂ φRational ≡ (n,d) ⦂ ⦃ num: ℤ, den: ℤ ⦄ 𝗌𝗎𝖻𝗃 n d. of_int n / of_int d = x ∧ d ≠ 0›
  deriving Basic
       and ‹Object_Equiv ℚ (=)›
       and ‹Abstract_Domain⇩L ℚ (λ_. True)›
       and ‹Abstract_Domain ℚ (λ_. True)›
       and Semantic_Type
       and Inhabited

  proc rat_add:
    input ‹𝗏𝖺𝗅 q1 ⦂ ℚ ❟ 𝗏𝖺𝗅 q2 ⦂ ℚ›
    output ‹𝗏𝖺𝗅 q1 + q2 ⦂ ℚ›
  ❴  
    val q1 = (q1 𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 𝗈𝗉𝖾𝗇) ；
    val q2 = (q2 𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 𝗈𝗉𝖾𝗇) ；
    ⦃ num: q1.num * q2.den + q2.num * q1.den,
      den: q1.den * q2.den ⦄
    𝗆𝖺𝗄𝖾𝗌 ‹ℚ›
  ❵ .


  proc rat_sub:
    input ‹q1 ⦂ 𝗏𝖺𝗅 ℚ ❟ q2 ⦂ 𝗏𝖺𝗅 ℚ›
    output ‹q1 - q2 ⦂ 𝗏𝖺𝗅 ℚ›
  ❴
    val q1 = (q1 𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 𝗈𝗉𝖾𝗇) ；
    val q2 = (q2 𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 𝗈𝗉𝖾𝗇) ；
    ⦃ num: q1.num * q2.den - q2.num * q1.den,
      den: q1.den * q2.den ⦄
    𝗆𝖺𝗄𝖾𝗌 ‹ℚ›
  ❵ .
  
  
  proc rat_mul:
    input ‹q1 ⦂ 𝗏𝖺𝗅 ℚ ❟ q2 ⦂ 𝗏𝖺𝗅 ℚ›
    output ‹q1 * q2 ⦂ 𝗏𝖺𝗅 ℚ›
  ❴  
    val q1 = (q1 𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 𝗈𝗉𝖾𝗇) ；
    val q2 = (q2 𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 𝗈𝗉𝖾𝗇) ；
    ⦃ num: q1.num * q2.num,
      den: q1.den * q2.den ⦄
    𝗆𝖺𝗄𝖾𝗌 ‹ℚ›
  ❵ .
  
  
  proc rat_div:
    input ‹𝗏𝖺𝗅 q1 ⦂ ℚ ❟ 𝗏𝖺𝗅 q2 ⦂ ℚ›
    premises ‹q2 ≠ 0›
    output ‹𝗏𝖺𝗅 q1 / q2 ⦂ ℚ›
  ❴  
    val q1 ← (q1 𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 𝗈𝗉𝖾𝗇) ；
    val q2 ← (q2 𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 𝗈𝗉𝖾𝗇) ；
    ⦃ num: q1.num * q2.den,
      den: q1.den * q2.num ⦄
    𝗆𝖺𝗄𝖾𝗌 ‹ℚ›
  ❵ .


proc rat_lt [φoverload <]:
  input ‹𝗏𝖺𝗅 q1 ⦂ ℚ❟ 𝗏𝖺𝗅 q2 ⦂ ℚ›
  output ‹q1 < q2 ⦂ 𝗏𝖺𝗅 𝔹›
❴
  val q1 ← (q1 𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 𝗈𝗉𝖾𝗇) ；
  val q2 ← (q2 𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 𝗈𝗉𝖾𝗇) ；
  val a ← q1.num * q2.den ；
  val b ← q1.den * q2.num ；
  sel (q1.den > 0 ⊕ q2.den > 0, a > b, a < b)
❵ .

proc rat_le [φoverload ≤]:
  input ‹𝗏𝖺𝗅 q1 ⦂ ℚ❟ 𝗏𝖺𝗅 q2 ⦂ ℚ›
  output ‹q1 ≤ q2 ⦂ 𝗏𝖺𝗅 𝔹›
❴
  val q1 ← (q1 𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 𝗈𝗉𝖾𝗇) ；
  val q2 ← (q2 𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 𝗈𝗉𝖾𝗇) ；
  val a ← q1.num * q2.den ；
  val b ← q1.den * q2.num ；
  sel (q1.den > 0 ⊕ q2.den > 0, a ≥ b, a ≤ b)
❵  .

proc rat_gt [φoverload >]:
  input ‹𝗏𝖺𝗅 q1 ⦂ ℚ❟ 𝗏𝖺𝗅 q2 ⦂ ℚ›
  output ‹q1 > q2 ⦂ 𝗏𝖺𝗅 𝔹›
❴
  q2 < q1
❵ .

proc rat_ge [φoverload >]:
  input ‹𝗏𝖺𝗅 q1 ⦂ ℚ❟ 𝗏𝖺𝗅 q2 ⦂ ℚ›
  output ‹q1 ≥ q2 ⦂ 𝗏𝖺𝗅 𝔹›
❴
  q2 ≤ q1
❵ .


text ‹The Conclusions of above Certification is the following Specification Theorems›

thm rat_add_φapp
thm rat_sub_φapp
thm rat_mul_φapp
thm rat_div_φapp

text ‹Semantic Representations of the Programs: ›

thm rat_add_def

end
