local

val is_poison_0 = @{lemma
  ‹ semty_ntup (fmupd k 𝗉𝗈𝗂𝗌𝗈𝗇 Ts) ≡ 𝗉𝗈𝗂𝗌𝗈𝗇 ›
  by (simp add: atomize_eq semty_ntup_eq_poison, meson fmranI fmupd_lookup) }

val is_poison_N = @{lemma
  ‹ semty_ntup Ts ≡ 𝗉𝗈𝗂𝗌𝗈𝗇
⟹ k |∉| fmdom Ts
⟹ semty_ntup (fmupd k TY Ts) ≡ 𝗉𝗈𝗂𝗌𝗈𝗇 ›
  by (clarsimp simp add: semty_ntup_eq_poison atomize_eq,
      metis fmdom_notD fmlookup_ran_iff fmupd_lookup option.distinct(1)) }

fun has_poison (Const(\<^const_name>‹fmempty›, _)) = SOME false
  | has_poison (Const(\<^const_name>‹fmupd›, _) $ _ $ Const(\<^const_name>‹𝗉𝗈𝗂𝗌𝗈𝗇›, _) $ _) = SOME true
  | has_poison (Const(\<^const_name>‹fmupd›, _) $ _ $ _ $ X) = has_poison X
  | has_poison _ = NONE

in

val setup_semty_ntup_to_poison =
  Simplifier.define_simproc \<^binding>‹semty_ntup_to_poison› {
      lhss = [\<^term>‹semty_ntup X›],
      proc = fn _ => fn ctxt => fn ctm =>
        case Thm.term_of ctm
          of Const(\<^const_name>‹semty_ntup›, _) $ X =>
              (case has_poison X
                 of SOME true =>
                      Thm.apply (Thm.apply \<^cterm>‹Pure.eq :: TY ⇒ TY ⇒ prop› ctm) \<^cterm>‹𝗉𝗈𝗂𝗌𝗈𝗇›
                        |> Thm.trivial
                        |> REPEAT_DETERM (CHANGED (resolve_tac ctxt [is_poison_0, is_poison_N] 1))
                        |> Seq.pull
                        |> Option.map fst
                  | _ => NONE)
           | _ => NONE
              
    }


end
