Theory Linked_List

theory Linked_List
  imports Phi_Semantics.PhiSem_C
begin

abbreviation ‹𝗅𝗂𝗇𝗄_𝗅𝗂𝗌𝗍 TY ≡ 𝗌𝗍𝗋𝗎𝖼𝗍 {nxt: 𝗉𝗍𝗋, data: TY}›

φtype_def Linked_Lst :: ‹address ⇒ (VAL, 'a) φ ⇒ (fiction, 'a list) φ›
  where ‹([] ⦂ Linked_Lst addr T)   = (Void 𝗌𝗎𝖻𝗃 addr = 0 ∧ 𝗍𝗒𝗉𝖾𝗈𝖿 T ≠ 𝗉𝗈𝗂𝗌𝗈𝗇)›
      | ‹(x#ls ⦂ Linked_Lst addr T) = (ls ⦂ Linked_Lst nxt T❟
                                      (nxt, x) ⦂ 𝗈𝖻𝗃[addr] ⦃ nxt: Ptr, data: T ⦄
                                      𝗌𝗎𝖻𝗃 nxt. 𝗍𝗒𝗉𝖾𝗈𝖿 addr = 𝗅𝗂𝗇𝗄_𝗅𝗂𝗌𝗍 (𝗍𝗒𝗉𝖾𝗈𝖿 T) )›

     deriving Basic
          and ‹Abstract_Domain T P ⟹ Abstract_Domain (Linked_Lst addr T)
                                      (λx. list_all P x ∧ (x = [] ⟷ addr = 0) ∧ 𝗍𝗒𝗉𝖾𝗈𝖿 T ≠ 𝗉𝗈𝗂𝗌𝗈𝗇) ›
          and ‹Identity_Elements⇩E (Linked_Lst addr T) (λl. addr = 0 ∧ l = [] ∧ 𝗍𝗒𝗉𝖾𝗈𝖿 T ≠ 𝗉𝗈𝗂𝗌𝗈𝗇)›
          and ‹Identity_Elements⇩I (Linked_Lst addr T) (λl. l = [] ∧ 𝗍𝗒𝗉𝖾𝗈𝖿 T ≠ 𝗉𝗈𝗂𝗌𝗈𝗇) (λ_. addr = 0)›
          and ‹ 𝗉𝗋𝖾𝗆𝗂𝗌𝖾 𝗍𝗒𝗉𝖾𝗈𝖿 T = 𝗍𝗒𝗉𝖾𝗈𝖿 U
            ⟹ Transformation_Functor (Linked_Lst addr) (Linked_Lst addr) T U set (λ_. UNIV) list_all2›
          and ‹ 𝗉𝗋𝖾𝗆𝗂𝗌𝖾 𝗍𝗒𝗉𝖾𝗈𝖿 T = 𝗍𝗒𝗉𝖾𝗈𝖿 U
            ⟹ Functional_Transformation_Functor (Linked_Lst addr) (Linked_Lst addr) T U
                                                   set (λx. UNIV) (λf. list_all) (λf P. map f)›
          and Pointer_Of


proc init:
  input  Void
  premises ‹𝗍𝗒𝗉𝖾𝗈𝖿 T ≠ 𝗉𝗈𝗂𝗌𝗈𝗇›
  output ‹[] ⦂ 𝗋𝖾𝖿 Linked_Lst 0 T›
❴
  𝗆𝖺𝗄𝖾𝗌(0) ‹Linked_Lst _ T› ；
  NULL
❵ .

proc is_empty:
  input  ‹l ⦂ 𝗋𝖾𝖿 Linked_Lst addr T›
  output ‹l = [] ⦂ 𝗏𝖺𝗅 𝔹❟ l ⦂ Linked_Lst addr T›
  is [routine]
❴
  addr = NULL
❵ .

  
proc prepend_llist:
  input  ‹l ⦂ 𝗋𝖾𝖿 Linked_Lst addr T❟ v ⦂ 𝗏𝖺𝗅 T›
  requires ‹Semantic_Zero_Val (𝗍𝗒𝗉𝖾𝗈𝖿 T) T z›
  output ‹v#l ⦂ 𝗋𝖾𝖿 Linked_Lst addr' T 𝗌𝗎𝖻𝗃 addr'. ⊤›
❴
  val ret ← calloc1 ‹⦃ nxt: Ptr, data: T ⦄› ；
  ret.nxt := addr ； 
  ret.data := v ；
  𝗆𝖺𝗄𝖾𝗌(1) ‹Linked_Lst _ T› ；
  ret
❵ .

proc pop_llist:
  input  ‹l ⦂ 𝗋𝖾𝖿 Linked_Lst addr T ›
  output ‹(if l = [] then [] else tl l) ⦂ 𝗋𝖾𝖿 Linked_Lst addr' T
          𝗌𝗎𝖻𝗃 addr'. ⊤›
  is [routine]
❴
  if (addr = NULL) ❴
    return(addr)
  ❵ ❴ ❵ ；
   
  𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 ‹𝗈𝗉𝖾𝗇(1)› ；
  val ret ← addr.nxt ；
  mfree (addr) ；

  ret  
❵ .


proc nth_llist:
  input    ‹l ⦂ 𝗋𝖾𝖿 Linked_Lst addr T❟ i ⦂ 𝗏𝖺𝗅 ℕ(𝗂𝗇𝗍)›
  premises ‹i < length l›
  output   ‹l ⦂ Linked_Lst addr T❟ l!i ⦂ 𝗏𝖺𝗅 T›
  is [recursive l i addr]
❴
  𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 ‹𝗈𝗉𝖾𝗇(1)› ；
  if (i = 0) ❴
      addr.data
  ❵ ❴
      nth_llist (addr.nxt, i - 1)
  ❵
  𝗆𝖺𝗄𝖾𝗌(1) ‹Linked_Lst addr T›
❵ .


proc hd_llist:
  input  ‹l ⦂ 𝗋𝖾𝖿 Linked_Lst addr T›
  premises ‹l ≠ []›
  output ‹hd l ⦂ 𝗏𝖺𝗅 T❟ l ⦂ Linked_Lst addr T›
❴
  nth_llist (addr, 0)
❵ .


proc update_nth_llist:
  input    ‹l ⦂ 𝗋𝖾𝖿 Linked_Lst addr T❟ i ⦂ 𝗏𝖺𝗅 ℕ(𝗂𝗇𝗍)❟ y ⦂ 𝗏𝖺𝗅 T›
  premises ‹i < length l›
  output   ‹l[i := y] ⦂ Linked_Lst addr T›
  is [recursive l i addr]
❴
  𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 ‹𝗈𝗉𝖾𝗇(1)› ；
  if (i = 0) ❴
      addr.data := y
  ❵ ❴
      update_nth_llist (addr.nxt, i - 1, y)
  ❵ ；
  𝗆𝖺𝗄𝖾𝗌(1) ‹Linked_Lst addr T›
❵ .


proc length_of:
  input    ‹l ⦂ 𝗋𝖾𝖿 Linked_Lst addr T›
  premises ‹length l < 2 ^ LENGTH(𝗂𝗇𝗍)›
  output   ‹length l ⦂ 𝗏𝖺𝗅 ℕ(𝗂𝗇𝗍)❟ l ⦂ Linked_Lst addr T›
  is [recursive l addr]
  is [routine]
❴
  if (addr = NULL) ❴
    return(0)
  ❵ ❴
    𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 ‹𝗈𝗉𝖾𝗇(1)› ；
    addr.nxt → val t1 ；
    val ret ← length_of (addr.nxt) + 1 ；
    𝗆𝖺𝗄𝖾𝗌(1) ‹Linked_Lst addr T› ； (* 1: call the second constructor *)
    return (ret)
  ❵
❵ .


proc reverse_aux:
  input  ‹l  ⦂ 𝗋𝖾𝖿 Linked_Lst addr T❟
          l' ⦂ 𝗋𝖾𝖿 Linked_Lst addr' T›
  output ‹rev l @ l' ⦂ 𝗋𝖾𝖿 Linked_Lst addr'' T
          𝗌𝗎𝖻𝗃 addr''. ⊤›
  is [recursive l l' addr addr']
❴
  if (addr = NULL) ❴
    𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 ‹𝗈𝗉𝖾𝗇(0)›
    addr'
  ❵ ❴
    𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 ‹𝗈𝗉𝖾𝗇(1)› ；
    addr.nxt → val aa ；
    addr.nxt := addr' ；
    ‹Linked_Lst addr' T› 𝗆𝖺𝗄𝖾𝗌(1) ‹hd l # l' ⦂ Linked_Lst addr T› ；
    reverse_aux (aa, addr)
  ❵
❵ .

proc reverse:
  input  ‹l ⦂ 𝗋𝖾𝖿 Linked_Lst addr T›
  output ‹rev l ⦂ 𝗋𝖾𝖿 Linked_Lst addr'  T
          𝗌𝗎𝖻𝗃 addr'. ⊤›
❴
  𝗆𝖺𝗄𝖾𝗌(0) ‹Linked_Lst 0 T›
  reverse_aux( addr, NULL )
            ― ‹^ φ-type annotation here is not considered as a manual annotation because
                  it is exactly the C language type of the pointer literal.It can be
                  automatically generated by an C parser. 0 denotes the NULL pointer. ›
❵ .
