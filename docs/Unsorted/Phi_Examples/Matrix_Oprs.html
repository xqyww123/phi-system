Theory Matrix_Oprs

theory Matrix_Oprs
  imports Phi_Semantics.PhiSem_C
          Jordan_Normal_Form.Matrix
          PhiStd.PhiStd_Slice
          Phi_Semantics.PhiSem_Mem_C_MI
begin

abbreviation ‹𝗆𝖺𝗍 M N ≡ 𝖺𝗋𝗋𝖺𝗒[M] 𝖺𝗋𝗋𝖺𝗒[N] 𝗂𝗇𝗍›
 
φtype_def MatSlice :: ‹address ⇒ nat ⇒ nat ⇒ nat ⇒ nat ⇒ (fiction, int mat) φ›
  where ‹x ⦂ MatSlice addr i j m n ≡ l ⦂ 𝗆𝖾𝗆[addr] 𝗌𝗅𝗂𝖼𝖾[i,m] (𝗌𝗅𝗂𝖼𝖾[j,n] ℤ⇧r(𝗂𝗇𝗍))
                                     𝗌𝗎𝖻𝗃 l. l = mat_to_list x ∧ x ∈ carrier_mat m n›
  parameter_equality (id,id,id,id,id)
  deriving ‹Abstract_Domain (MatSlice addr i j m n) (λx. addr ≠ 0 ∧ x ∈ carrier_mat m n)›
       and ‹Object_Equiv (MatSlice addr i j m n) (=)›
       and Basic



lemmas [φsledgehammer_simps] = mat_to_list_def list_eq_iff_nth_eq list_all2_conv_all_nth zero_mat_def

proc zero_mat:
  input    ‹a ⦂ 𝗏𝖺𝗅 Ptr[𝗆𝖺𝗍 M N]❟ i ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟ j ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟
            m ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟ n ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟
            x ⦂ MatSlice a i j m n›
  premises ‹i + m ≤ M ∧ j + n ≤ N ∧ M < 2 ^ addrspace_bits ∧ N < 2 ^ addrspace_bits›
  output   ‹0⇩m m n ⦂ MatSlice a i j m n›
  unfolding MatSlice.unfold
❴
  map_slice(m) ❴ for k → val k ；
    map_slice(n) ❴ for h → val h  ；
      a[i+k, j+h] := ‹0 ⦂ ℤ⇧r(𝗂𝗇𝗍)›
    ❵
  ❵ ；

❵ .


proc new_mat:
  requires ‹𝗉𝖺𝗋𝖺𝗆 M›
  requires ‹𝗉𝖺𝗋𝖺𝗆 N›
  input    ‹Void›
  output   ‹a ⦂ 𝗏𝖺𝗅 Ptr[𝗆𝖺𝗍 M N]❟
            0⇩m M N ⦂ MatSlice a 0 0 M N
            𝗌𝗎𝖻𝗃 a. address_to_base a›
  unfolding MatSlice.unfold
❴
  calloc1 ‹𝔸𝗋𝗋𝖺𝗒[M] 𝔸𝗋𝗋𝖺𝗒[N] ℤ⇧r(𝗂𝗇𝗍)›
❵ .

proc del_mat:
  input    ‹a ⦂ 𝗏𝖺𝗅 Ptr[𝗆𝖺𝗍 M N]❟ x ⦂ MatSlice a i j m n›
  premises ‹m = M ∧ n = N ∧ i = 0 ∧ j = 0 ∧ address_to_base a›
  output   ‹Void›
  unfolding MatSlice.unfold
❴
  mfree (a)
❵ .


proc copy_mat:
  input    ‹a⇩x ⦂ 𝗏𝖺𝗅 Ptr[𝗆𝖺𝗍 M⇩x N⇩x]❟ i⇩x ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟ j⇩x ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟
            a⇩y ⦂ 𝗏𝖺𝗅 Ptr[𝗆𝖺𝗍 M⇩y N⇩y]❟ i⇩y ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟ j⇩y ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟
            m ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟ n ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟
            x ⦂ MatSlice a⇩x i⇩x j⇩x m n❟
            y ⦂ MatSlice a⇩y i⇩y j⇩y m n›

  premises ‹i⇩x + m ≤ M⇩x ∧ i⇩y + m ≤ M⇩y ∧ j⇩x + n ≤ N⇩x ∧ j⇩y + n ≤ N⇩y ∧
            M⇩x < 2 ^ addrspace_bits ∧ N⇩x < 2 ^ addrspace_bits ∧
            M⇩y < 2 ^ addrspace_bits ∧ N⇩y < 2 ^ addrspace_bits›

  output   ‹y ⦂ MatSlice a⇩x i⇩x j⇩x m n❟ y ⦂ MatSlice a⇩y i⇩y j⇩y m n›
  unfolding MatSlice.unfold
❴
  map_2slice (m) ❴ for k → val k ；
    map_2slice (n) ❴ for h → val h ；
      a⇩x[i⇩x+k, j⇩x+h] := a⇩y[i⇩y+k, j⇩y+h]
    ❵ ；
  ❵
❵ .


proc add_mat:
  input    ‹a⇩x ⦂ 𝗏𝖺𝗅 Ptr[𝗆𝖺𝗍 M⇩x N⇩x]❟ i⇩x ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟ j⇩x ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟
            a⇩y ⦂ 𝗏𝖺𝗅 Ptr[𝗆𝖺𝗍 M⇩y N⇩y]❟ i⇩y ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟ j⇩y ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟
            m ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟ n ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟
            x ⦂ MatSlice a⇩x i⇩x j⇩x m n❟
            y ⦂ MatSlice a⇩y i⇩y j⇩y m n›

  premises ‹i⇩x + m ≤ M⇩x ∧ i⇩y + m ≤ M⇩y ∧ j⇩x + n ≤ N⇩x ∧ j⇩y + n ≤ N⇩y ∧
            M⇩x < 2 ^ addrspace_bits ∧ N⇩x < 2 ^ addrspace_bits ∧
            M⇩y < 2 ^ addrspace_bits ∧ N⇩y < 2 ^ addrspace_bits›

  output   ‹x + y ⦂ MatSlice a⇩x i⇩x j⇩x m n❟ y ⦂ MatSlice a⇩y i⇩y j⇩y m n›
  unfolding MatSlice.unfold
❴
  map_2slice (m) ❴ for k → val k
    map_2slice (n) ❴ for h → val h
        a⇩x[i⇩x+k, j⇩x+h] := a⇩x[i⇩x+k, j⇩x+h] + a⇩y[i⇩y+k, j⇩y+h]
    ❵ ；
  ❵
❵ .


proc sub_mat:
  input    ‹a⇩x ⦂ 𝗏𝖺𝗅 Ptr[𝗆𝖺𝗍 M⇩x N⇩x]❟ i⇩x ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟ j⇩x ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟
            a⇩y ⦂ 𝗏𝖺𝗅 Ptr[𝗆𝖺𝗍 M⇩y N⇩y]❟ i⇩y ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟ j⇩y ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟
            m ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟ n ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟
            x ⦂ MatSlice a⇩x i⇩x j⇩x m n❟
            y ⦂ MatSlice a⇩y i⇩y j⇩y m n›

  premises ‹i⇩x + m ≤ M⇩x ∧ i⇩y + m ≤ M⇩y ∧ j⇩x + n ≤ N⇩x ∧ j⇩y + n ≤ N⇩y ∧
            M⇩x < 2 ^ addrspace_bits ∧ N⇩x < 2 ^ addrspace_bits ∧
            M⇩y < 2 ^ addrspace_bits ∧ N⇩y < 2 ^ addrspace_bits ›

  output   ‹x - y ⦂ MatSlice a⇩x i⇩x j⇩x m n❟ y ⦂ MatSlice a⇩y i⇩y j⇩y m n›
  unfolding MatSlice.unfold
❴
  map_2slice (m) ❴ for k → val k ；
    map_2slice (n) ❴ for h → val h ；
      a⇩x[i⇩x+k, j⇩x+h] := a⇩x[i⇩x+k, j⇩x+h] - a⇩y[i⇩y+k, j⇩y+h]
    ❵ ；
  ❵ ；
❵ .


context
  notes [φsledgehammer_simps] = Let_def image_iff split_block_def four_block_mat_def mat_to_list_def
                                list_eq_iff_nth_eq nth_append Bex_def in_set_conv_nth image_iff
begin

lemma split_4mat:
  ‹ 𝗉𝖺𝗋𝖺𝗆 (s, t)
⟹ 𝗉𝗋𝖾𝗆𝗂𝗌𝖾 s ≤ m ∧ t ≤ n
⟹ x ⦂ MatSlice a i j m n 𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌 x⇩1⇩1 ⦂ MatSlice a i j s t❟ x⇩1⇩2 ⦂ MatSlice a i (j+t) s (n-t)❟
                                    x⇩2⇩1 ⦂ MatSlice a (i+s) j (m-s) t❟ x⇩2⇩2 ⦂ MatSlice a (i+s) (j+t) (m-s) (n-t)
                                    𝗌𝗎𝖻𝗃 x⇩1⇩1 x⇩1⇩2 x⇩2⇩1 x⇩2⇩2. (x⇩1⇩1, x⇩1⇩2, x⇩2⇩1, x⇩2⇩2) = split_block x s t›
  unfolding MatSlice.unfold ― ‹open abstraction in both sides›
  ❴ ❵ certified by (auto simp: Bex_def in_set_conv_nth image_iff; auto_sledgehammer) .

lemma merge_4mat:
  ‹ 𝗉𝗋𝖾𝗆𝗂𝗌𝖾 s ≤ m ∧ t ≤ n
⟹ x⇩1⇩1 ⦂ MatSlice a i j s t❟ x⇩1⇩2 ⦂ MatSlice a i (j+t) s (n-t)❟
    x⇩2⇩1 ⦂ MatSlice a (i+s) j (m-s) t❟ x⇩2⇩2 ⦂ MatSlice a (i+s) (j+t) (m-s) (n-t)
    𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌 four_block_mat x⇩1⇩1 x⇩1⇩2 x⇩2⇩1 x⇩2⇩2 ⦂ MatSlice a i j m n ›
  unfolding MatSlice.unfold ― ‹open abstraction in both sides›
  ❴ ❵ certified by (auto simp: φsledgehammer_simps φ[unfolded carrier_mat_def, simplified]) .

end


declare [[linarith_split_limit = 20]]

context
  fixes N :: nat
begin
 
proc strassen:
  input    ‹a⇩A ⦂ 𝗏𝖺𝗅 Ptr[𝗆𝖺𝗍 N N]❟ i⇩A ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟ j⇩A ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟
            a⇩B ⦂ 𝗏𝖺𝗅 Ptr[𝗆𝖺𝗍 N N]❟ i⇩B ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟ j⇩B ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟
            a⇩C ⦂ 𝗏𝖺𝗅 Ptr[𝗆𝖺𝗍 N N]❟ i⇩C ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟ j⇩C ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟
            a⇩D ⦂ 𝗏𝖺𝗅 Ptr[𝗆𝖺𝗍 N N]❟ i⇩D ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟ j⇩D ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟
            n ⦂ 𝗏𝖺𝗅 ℕ(𝗌𝗂𝗓𝖾_𝗍)❟
            A ⦂ MatSlice a⇩A i⇩A j⇩A (2^n) (2^n)❟
            B ⦂ MatSlice a⇩B i⇩B j⇩B (2^n) (2^n)❟
            buf⇩1 ⦂ MatSlice a⇩C i⇩C j⇩C (2^n) (2^n)❟ ― ‹buffers used to store intermediate values›
            buf⇩2 ⦂ MatSlice a⇩D i⇩D j⇩D (2^n) (2^n)›

  premises ‹i⇩A + 2^n ≤ N ∧ j⇩A + 2^n ≤ N ∧
            i⇩B + 2^n ≤ N ∧ j⇩B + 2^n ≤ N ∧
            i⇩C + 2^n ≤ N ∧ j⇩C + 2^n ≤ N ∧
            i⇩D + 2^n ≤ N ∧ j⇩D + 2^n ≤ N ∧
            M < 2 ^ addrspace_bits ∧ N < 2 ^ addrspace_bits›

  output   ‹A * B ⦂ MatSlice a⇩A i⇩A j⇩A (2^n) (2^n)❟
            B ⦂ MatSlice a⇩B i⇩B j⇩B (2^n) (2^n)❟
            buf⇩1 ⦂ MatSlice a⇩C i⇩C j⇩C (2^n) (2^n)❟ ― ‹buffers used to store intermediate values›
            buf⇩2 ⦂ MatSlice a⇩D i⇩D j⇩D (2^n) (2^n)
            𝗌𝗎𝖻𝗃 buf⇩1 buf⇩2. ⊤›
  is [recursive]
❴
  if (n = 0)
  ❴
    ‹MatSlice a⇩A _ _ _ _› 𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 𝗈𝗉𝖾𝗇
    ‹MatSlice a⇩B _ _ _ _› 𝗍𝗋𝖺𝗇𝗌𝖿𝗈𝗋𝗆𝗌_𝗍𝗈 𝗈𝗉𝖾𝗇  ；

    a⇩A[i⇩A, j⇩A] := a⇩A[i⇩A, j⇩A] * a⇩B[i⇩B, j⇩B] ；
      
    holds_fact [φsledgehammer_simps]: ‹dim_col B = 1 ∧ dim_row B = 1 ∧ dim_col A = 1 ∧ dim_row A = 1› (*for proof obligation*)
    note scalar_prod_def [φsledgehammer_simps] ；                                       (*for proof obligation*)

    𝗆𝖺𝗄𝖾𝗌 ‹A * B ⦂ MatSlice a⇩A i⇩A j⇩A (2^n) (2^n)› ；
    𝗆𝖺𝗄𝖾𝗌 ‹B ⦂ MatSlice a⇩B i⇩B j⇩B (2^n) (2^n)›
  ❵
  ❴
    holds_fact t0: ‹(2::nat) ^ n = 2 ^ (n - 1) + 2 ^ (n - 1)›                          (*for proof obligation*)
    holds_fact t1[simp]: ‹(2::nat) ^ n - 2 ^ (n - 1) = 2 ^ (n - 1)› ；                  (*an annotation for reasoning*)
        ― ‹Even when ML-aided ATP ‹Sledgehammer› is so powerful now, it cannot solve some simple
            high-school problems without the ‹t0› hint ...›

    ‹MatSlice a⇩A _ _ _ _› split_4mat ‹(2^(n-1), 2^(n-1))› ∃A⇩1⇩1, A⇩1⇩2, A⇩2⇩1, A⇩2⇩2 ；
    ‹MatSlice a⇩B _ _ _ _› split_4mat ‹(2^(n-1), 2^(n-1))› ∃B⇩1⇩1, B⇩1⇩2, B⇩2⇩1, B⇩2⇩2 ；
    ‹MatSlice a⇩C _ _ _ _› split_4mat ‹(2^(n-1), 2^(n-1))› ∃M⇩1, M⇩2, M⇩3, M⇩4 ；
    ‹MatSlice a⇩D _ _ _ _› split_4mat ‹(2^(n-1), 2^(n-1))› ∃M⇩5, M⇩6, M⇩7, M⇩t ；

    holds_fact t2: ‹n < addrspace_bits› ；                                              (*for proof obligation*)

    1 << (n-1) → val N ；
    i⇩A + N → val i⇩A' ；
    j⇩A + N → val j⇩A' ；
    i⇩B + N → val i⇩B' ；
    j⇩B + N → val j⇩B' ；
    i⇩C + N → val i⇩C' ；
    j⇩C + N → val j⇩C' ；
    i⇩D + N → val i⇩D' ；
    j⇩D + N → val j⇩D' ；

    copy_mat (a⇩C, i⇩C , j⇩C,  (*M⇩1*) a⇩A, i⇩A , j⇩A , (*A⇩1⇩1*) N, N) ；
    add_mat  (a⇩C, i⇩C , j⇩C,  (*M⇩1*) a⇩A, i⇩A', j⇩A', (*A⇩2⇩2*) N, N) ；
    copy_mat (a⇩D, i⇩D', j⇩D', (*M⇩t*) a⇩B, i⇩B , j⇩B , (*B⇩1⇩1*) N, N) ；
    add_mat  (a⇩D, i⇩D', j⇩D', (*M⇩t*) a⇩B, i⇩B', j⇩B', (*B⇩2⇩2*) N, N) ；
    strassen (a⇩C, i⇩C , j⇩C , (*M⇩1*) a⇩D, i⇩D', j⇩D', (*M⇩t *)
              a⇩C, i⇩C , j⇩C', (*M⇩2*) a⇩C, i⇩C', j⇩C , (*M⇩3 *) n-1) ；

    copy_mat (a⇩C, i⇩C, j⇩C', (*M⇩2*) a⇩A, i⇩A', j⇩A , (*A⇩2⇩1*) N, N) ；
    add_mat  (a⇩C, i⇩C, j⇩C', (*M⇩2*) a⇩A, i⇩A', j⇩A', (*A⇩2⇩2*) N, N) ；
    strassen (a⇩C, i⇩C, j⇩C', (*M⇩2*) a⇩B, i⇩B , j⇩B , (*B⇩1⇩1*)
              a⇩C, i⇩C', j⇩C, (*M⇩3*) a⇩C, i⇩C', j⇩C', (*M⇩4*) n-1) ；

    copy_mat (a⇩D, i⇩D', j⇩D', (*M⇩t*) a⇩B, i⇩B , j⇩B', (*B⇩1⇩2*) N, N) ；
    sub_mat  (a⇩D, i⇩D', j⇩D', (*M⇩t*) a⇩B, i⇩B', j⇩B', (*B⇩2⇩2*) N, N) ；
    copy_mat (a⇩C, i⇩C', j⇩C , (*M⇩3*) a⇩A, i⇩A , j⇩A , (*A⇩1⇩1*) N, N) ；
    strassen (a⇩C, i⇩C', j⇩C , (*M⇩3*) a⇩D, i⇩D', j⇩D', (*M⇩t*)
              a⇩C, i⇩C', j⇩C', (*M⇩4*) a⇩D, i⇩D , j⇩D , (*M⇩5*) n-1) ；


    copy_mat (a⇩D, i⇩D', j⇩D', (*M⇩t*) a⇩B, i⇩B', j⇩B , (*B⇩2⇩1*) N, N) ；
    sub_mat  (a⇩D, i⇩D', j⇩D', (*M⇩t*) a⇩B, i⇩B , j⇩B , (*B⇩1⇩1*) N, N) ；
    copy_mat (a⇩C, i⇩C', j⇩C', (*M⇩4*) a⇩A, i⇩A', j⇩A', (*A⇩2⇩2*) N, N) ；
