Theory Binary_Search

theory Binary_Search
  imports Phi_Semantics.PhiSem_C
begin


proc binary_search_array:
  input  ‹arr ⦂ 𝗆𝖾𝗆[ptr] 𝔸𝗋𝗋𝖺𝗒[cap] ℕ(𝗂𝗇𝗍)❟
          ptr ⦂ 𝗏𝖺𝗅 Ptr[𝖺𝗋𝗋𝖺𝗒[cap] 𝗂𝗇𝗍]❟ lower ⦂ 𝗏𝖺𝗅 ℕ(𝗂𝗇𝗍)❟ upper ⦂ 𝗏𝖺𝗅 ℕ(𝗂𝗇𝗍)❟ k ⦂ 𝗏𝖺𝗅 ℕ(𝗂𝗇𝗍)›
  premises ‹arr ! upper ≤ k›
       and ‹lower < upper›
       and ‹upper < cap›
       and ‹sorted arr›
  output ‹arr ⦂ 𝗆𝖾𝗆[ptr] 𝔸𝗋𝗋𝖺𝗒[cap] ℕ(𝗂𝗇𝗍)❟
          (LEAST i. lower ≤ i ∧ i ≤ upper ∧ arr!i ≤ k) ⦂ 𝗏𝖺𝗅 ℕ(𝗂𝗇𝗍)›
  is [routine]
❴
  if (ptr[lower] ≤ k) ❴
    return (lower)
  ❵ ❴
    (lower, upper) → var l, u
    while ‹l ⦂ 𝗏𝖺𝗋[l] ℕ(𝗂𝗇𝗍)❟ u ⦂ 𝗏𝖺𝗋[u] ℕ(𝗂𝗇𝗍) 𝗌𝗎𝖻𝗃 l u.
            Inv: (lower ≤ l ∧ l < u ∧ u ≤ upper ∧ k < arr!l ∧ arr!u ≤ k) ∧
            Guard: (l + 1 < u) ∧
            End: (l + 1 = u)›
         ( l + 1 < u )
    ❴
      val m ← l + (u - l) / 2 ；
      if ( ptr[m] ≤ k ) ❴ m → u ❵ ❴ m → l ❵
    ❵
    return (u)
  ❵
❵ .


proc generalized_binary_search:
  requires F: ‹∀i v. 𝗉𝗋𝗈𝖼 F v ⦃ i ⦂ 𝗏𝖺𝗅[v] ℕ(𝗂𝗇𝗍) ⟼ f i ⦂ 𝗏𝖺𝗅 𝔹 ⦄› ― ‹v: raw value›
  premises ‹∀i j. i ≤ j ⟶ f i ⟶ f j›
  input  ‹lower ⦂ 𝗏𝖺𝗅 ℕ(𝗂𝗇𝗍)❟ upper ⦂ 𝗏𝖺𝗅 ℕ(𝗂𝗇𝗍)›
  premises ‹f upper› and ‹lower < upper›
  output ‹(LEAST i. lower ≤ i ∧ i ≤ upper ∧ f i) ⦂ 𝗏𝖺𝗅 ℕ(𝗂𝗇𝗍)›
  is [routine]
❴

  if ( F(lower) ) ❴
     return (lower)
  ❵ ❴
    (lower, upper) → var l, u ;;
    while ‹l ⦂ 𝗏𝖺𝗋[l] ℕ(𝗂𝗇𝗍)❟ u ⦂ 𝗏𝖺𝗋[u] ℕ(𝗂𝗇𝗍) 𝗌𝗎𝖻𝗃 l u.
            Inv: (lower ≤ l ∧ l < u ∧ u ≤ upper ∧ ¬ f l ∧ f u) ∧
            Guard: (l + 1 < u) ∧
            End: (l + 1 = u)›
          ( ‹$l + 1 < $u› )
    ❴
      val m ← l + (u - l) / 2 ;;
      if ( F(m) ) ❴ m → u ❵ ❴ m → l ❵
    ❵
    return (u)
  ❵
❵.

text ‹The Conclusions of above Certification is the following Specification Theorems›

thm binary_search_array_φapp
thm generalized_binary_search_φapp

text ‹Semantic Representations of the Programs: ›

thm binary_search_array_def
thm generalized_binary_search_def

end
