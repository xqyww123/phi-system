                and sem_mk_tup   :: ‹VAL list ⇒ VAL›
                and sem_dest_tup :: ‹VAL ⇒ VAL list›
                where sem_dest_mk_tup[simp]: ‹sem_dest_tup (sem_mk_tup vs) = vs›
  and   WT_tup[simp]: ‹Well_Type (sem_tup_T ts)  = { sem_mk_tup vs |vs. list_all2 (λ t v. v ∈ Well_Type t) ts vs }›
  and   semty_tup_eq_poison[simp]: ‹sem_tup_T Ts = 𝗉𝗈𝗂𝗌𝗈𝗇 ⟷ 𝗉𝗈𝗂𝗌𝗈𝗇 ∈ set Ts›
  and   semty_tup_uniq:
            ‹sem_mk_tup vs ∈ Well_Type TY ⟹ ∃Ts. TY = sem_tup_T Ts›

  and   zero_tup[simp]: ‹sem_tup_T Ts ≠ 𝗉𝗈𝗂𝗌𝗈𝗇 ⟹ Zero (sem_tup_T Ts) = map_option sem_mk_tup (those (map Zero Ts))›
  and   V_tup_sep_disj_R[simp]: ‹sem_mk_tup l1 ## vl2 ⟷ (∃l2. vl2 = sem_mk_tup l2)›
  and   V_tup_sep_disj_L[simp]: ‹vl1 ## sem_mk_tup l2 ⟷ (∃l1. vl1 = sem_mk_tup l1)›
  and   V_tup_mult    : ‹sem_mk_tup l1 * sem_mk_tup l2 = sem_mk_tup (l1 @ l2)›
  and   idx_step_type_tup [eval_aggregate_path] : ‹ sem_tup_T tys ≠ 𝗉𝗈𝗂𝗌𝗈𝗇 ∧ i < length tys
                                                ⟹ idx_step_type (AgIdx_N i) (sem_tup_T tys) = tys!i ›
  and   valid_idx_step_tup[eval_aggregate_path] : ‹ sem_tup_T tys ≠ 𝗉𝗈𝗂𝗌𝗈𝗇
                                                ⟹ valid_idx_step (sem_tup_T tys) j ⟷ j ∈ {AgIdx_N i | i. i < length tys} ›
  and   idx_step_value_tup[eval_aggregate_path] : ‹idx_step_value (AgIdx_N i) (sem_mk_tup vs) = vs!i›
  and   idx_step_mod_value_tup : ‹idx_step_mod_value (AgIdx_N i) f (sem_mk_tup vs) = sem_mk_tup (vs[i := f (vs!i)])›

lemma Tuple_Type_Helper_S:
  ‹ Tuple_Type_Helper Ts TYs
⟹ Tuple_Type_Helper (⦃ T ⦄ ∗ Ts) (𝗍𝗒𝗉𝖾𝗈𝖿 T # TYs)  ›
  unfolding Tuple_Type_Helper_def SType_Of_def Inhabited_def Satisfiable_def Semantic_Type_def
  
  apply (auto)
  apply (metis V_tup_sep_disj_L)
  subgoal premises prems for x TY p xa TYa pa xb TYb pb proof -
    obtain TXX where t1: ‹TY = sem_tup_T TXX›
      using prems(1) prems(3) prems(9) semty_tup_uniq by blast
    show ?thesis apply (rule exI[where x=‹sem_tup_T (TYa # TXX)›]; auto)
      by (smt (verit, best) NO_MATCH_def V_tup_mult_cons WT_tup list_all2_Cons mem_Collect_eq prems(3) prems(5) t1)
  qed
  apply (smt (verit) Well_Type_poison someI zero_set_def zero_set_iff)
         apply (smt (verit, ccfv_threshold) NO_MATCH_def V_tup_mult_cons list.rel_inject(2) someI)

  subgoal premises prems for x TY p TYa a b xa l2 proof -
    obtain TH TXX where t1: ‹TYa = sem_tup_T (TH # TXX)›
      by (smt (z3) NO_MATCH_def V_tup_mult_cons V_tup_sep_disj_L WT_tup list_all2_Cons2 mem_Collect_eq prems(1) prems(4) prems(6) prems(8) sem_mk_tup_inj semty_tup_uniq)
    obtain xx vv where t2: ‹vv ⊨ (xx ⦂ T) ∧ vv ∉ Well_Type TH›
      using prems(5) by blast
    show ?thesis
      apply (insert ‹∀a b v.  _›[THEN spec[where x=xx], THEN spec[where x=b], THEN spec[where x=‹sem_mk_tup (vv # l2)›]],
          auto)
      apply (metis NO_MATCH_def V_tup_mult_cons V_tup_sep_disj_L prems(7) t2)
      by (simp add: t1 t2)
  qed

  using V_tup_mult apply force
  apply (smt (verit, del_insts) Well_Type_poison ex_in_conv someI_ex)

  subgoal premises prems for x TY p a TYa b cb xa proof -
    obtain cb' where t1: ‹cb = sem_mk_tup cb'›
      using prems(1) prems(7) by presburger
    have t2: ‹∃ca cb. sem_mk_tup (xa # cb') = ca * cb ∧ cb ⊨ (b ⦂ Ts) ∧ (∃x. ca = sem_mk_tup [x] ∧ x ⊨ (a ⦂ T)) ∧ ca ## cb›
      by (metis NO_MATCH_def V_tup_mult_cons V_tup_sep_disj_L prems(7) prems(8) t1)
    from ‹∀a b v. _›[THEN spec[where x=a], THEN spec[where x=b], THEN spec[where x=‹sem_mk_tup (xa # cb')›], THEN mp, OF t2]
    have t3: ‹sem_mk_tup (xa # cb') ∈ Well_Type TYa› .
    from semty_tup_uniq[OF this]
    obtain TH TXX where t1: ‹TYa = sem_tup_T (TH # TXX)›
      by (smt (verit, del_insts) WT_tup list_all2_Cons2 mem_Collect_eq sem_dest_mk_tup t3)

    show ?thesis apply (rule exI[where x=‹sem_tup_T TXX›]; auto)
      by (smt (z3) NO_MATCH_def V_tup_mult_cons V_tup_sep_disj_L WT_tup list_all2_Cons mem_Collect_eq prems(1) prems(6) prems(8) sem_dest_mk_tup t1)
  qed
  apply (metis V_tup_mult)

  subgoal premises prems for TY a b cb x proof -
    obtain cb' where t1: ‹cb = sem_mk_tup cb'›
      using prems(1) prems(6) by presburger
    obtain TTT where t2: ‹TY = sem_tup_T TTT›
      by (metis V_tup_mult V_tup_sep_disj_L prems(5) prems(6) prems(7) semty_tup_uniq t1)

    obtain x1 v1 where t3: ‹v1 ⊨ (x1 ⦂ T ) ∧ v1 ∉ Well_Type (hd TTT)›
      using prems(4) by blast
    obtain x2 v2 where t4: ‹sem_mk_tup v2 ⊨ (x2 ⦂ Ts) ∧ sem_mk_tup v2 ∉ Well_Type (sem_tup_T (tl TTT))›
      using prems(1) prems(3) by blast

    show ?thesis
      apply (insert ‹∀a b v. _›[THEN spec[where x=a], THEN spec[where x=b], THEN spec[where x=‹sem_mk_tup (v1 # v2)›]]; auto)
      apply (smt (z3) NO_MATCH_def V_tup_mult_cons V_tup_sep_disj_R WT_tup ‹⋀thesis. (⋀x2 v2. sem_mk_tup v2 ⊨ (x2 ⦂ Ts) ∧ sem_mk_tup v2 ∉ Well_Type (sem_tup_T (tl TTT)) ⟹ thesis) ⟹ thesis› list.sel(3) list_all2_Cons2 mem_Collect_eq prems(5) prems(7) sem_mk_tup_inj t2)
      by (smt (verit, best) WT_tup list.sel(3) list_all2_Cons2 mem_Collect_eq sem_dest_mk_tup t2 t4)
  qed
  using V_tup_mult by blast
