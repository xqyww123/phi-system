section ‹Semantics›

debt_axiomatization 𝖺𝗋𝖾𝖺𝗅 :: TY
                and sem_mk_areal  :: ‹real ⇒ VAL›
                and sem_dest_areal :: ‹VAL ⇒ real›
where sem_dest_mk_areal[simp]: ‹sem_dest_areal (sem_mk_areal i) = i›
  and WT_areal[simp]: ‹Well_Type 𝖺𝗋𝖾𝖺𝗅 = { sem_mk_areal i |i. True } ›
  and can_eqcmp_areal[simp]: "Can_EqCompare res (sem_mk_areal i1) (sem_mk_areal i2)"
  and eqcmp_areal[simp]: "EqCompare (sem_mk_areal i1) (sem_mk_areal i2) ⟷ i1 = i2"
  and  zero_areal[simp]: ‹Zero 𝖺𝗋𝖾𝖺𝗅 = Some (sem_mk_areal 0)›

lemma sem_mk_areal_inj[simp]:
  ‹sem_mk_areal i = sem_mk_areal j ⟷ i = j›
  by (metis sem_dest_mk_areal)

       and Inhabited

    "φEqual ℝ (λx y. True) (=)"
