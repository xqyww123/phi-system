local

val is_poison_0 = @{lemma
  ‹ sem_tup_T (𝗉𝗈𝗂𝗌𝗈𝗇 # Ts) ≡ 𝗉𝗈𝗂𝗌𝗈𝗇 ›
  by (smt (verit, ccfv_SIG) list.set_intros(1) semty_tup_eq_poison) }

val is_poison_N = @{lemma
  ‹ sem_tup_T Ts ≡ 𝗉𝗈𝗂𝗌𝗈𝗇
⟹ sem_tup_T (T # Ts) ≡ 𝗉𝗈𝗂𝗌𝗈𝗇 ›
  by (smt (verit, best) list.set_intros(2) semty_tup_eq_poison) }

fun has_poison (Const(\<^const_name>‹Nil›, _)) = SOME false
  | has_poison (Const(\<^const_name>‹Cons›, _) $ Const(\<^const_name>‹𝗉𝗈𝗂𝗌𝗈𝗇›, _) $ _) = SOME true
  | has_poison (Const(\<^const_name>‹Cons›, _) $ _ $ X) = has_poison X
  | has_poison _ = NONE

in

val setup_semty_tup_to_poison =
  Simplifier.define_simproc \<^binding>‹semty_ntup_to_poison› {
      lhss = [\<^term>‹sem_tup_T X›],
      proc = fn _ => fn ctxt => fn ctm =>
        case Thm.term_of ctm
          of Const(\<^const_name>‹sem_tup_T›, _) $ X =>
              (case has_poison X
                 of SOME true =>
                      Thm.apply (Thm.apply \<^cterm>‹Pure.eq :: TY ⇒ TY ⇒ prop› ctm) \<^cterm>‹𝗉𝗈𝗂𝗌𝗈𝗇›
                        |> Thm.trivial
                        |> REPEAT_DETERM (CHANGED (resolve_tac ctxt [is_poison_0, is_poison_N] 1))
                        |> Seq.pull
                        |> Option.map fst
                  | _ => NONE)
           | _ => NONE
              
    }


end
