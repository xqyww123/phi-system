debt_axiomatization mk_array_T :: ‹nat ⇒ TY ⇒ TY› ("𝖺𝗋𝗋𝖺𝗒[_] _" [20, 910] 910)
                and sem_mk_array   :: ‹VAL list ⇒ VAL›
                and sem_dest_array :: ‹VAL ⇒ VAL list›
  where sem_mk_dest_array[simp]: ‹sem_dest_array (sem_mk_array vs) = vs›
  and   semty_array_eq_poison[simp]: ‹𝖺𝗋𝗋𝖺𝗒[N] T = 𝗉𝗈𝗂𝗌𝗈𝗇 ⟷ (T = 𝗉𝗈𝗂𝗌𝗈𝗇 ∧ N ≠ 0)›
  and   WT_arr[simp]:   ‹Well_Type (𝖺𝗋𝗋𝖺𝗒[n] t) = { sem_mk_array vs |vs. length vs = n ∧ list_all (λv. v ∈ Well_Type t) vs }›
  and   semty_arr_uniq: ‹sem_mk_array vs ∈ Well_Type TY ⟹ ∃T. TY = mk_array_T (length vs) T›
  and   zero_arr[simp]: ‹𝖺𝗋𝗋𝖺𝗒[N] T ≠ 𝗉𝗈𝗂𝗌𝗈𝗇 ⟹
            Zero (𝖺𝗋𝗋𝖺𝗒[N] T) = (if N = 0 then Some (sem_mk_array [])
                                         else map_option (λz. sem_mk_array (replicate N z)) (Zero T))›
  and   idx_step_type_arr [eval_aggregate_path] : ‹T ≠ 𝗉𝗈𝗂𝗌𝗈𝗇 ∧ N ≠ 0 ⟹ idx_step_type (AgIdx_N i) (𝖺𝗋𝗋𝖺𝗒[N] T) = T›
  and   valid_idx_step_arr[eval_aggregate_path] : ‹T ≠ 𝗉𝗈𝗂𝗌𝗈𝗇 ⟹ valid_idx_step (𝖺𝗋𝗋𝖺𝗒[N] T) j ⟷ j ∈ {AgIdx_N i | i. i < N}›
  and   idx_step_value_arr[eval_aggregate_path] : ‹idx_step_value (AgIdx_N i) (sem_mk_array vs) = vs!i›
  and   idx_step_mod_value_arr : ‹idx_step_mod_value (AgIdx_N i) f (sem_mk_array vs) = sem_mk_array (vs[i := f (vs!i)])›
  and   type_measure_arr : ‹type_measure (𝖺𝗋𝗋𝖺𝗒[N] T) = (if T = 𝗉𝗈𝗂𝗌𝗈𝗇 ∨ N = 0 then 0 else Suc (type_measure T))›
  and   V_arr_sep_disj[simp]:
            ‹sem_mk_array vs1 ## sem_mk_array vs2 ⟷ list_all2 (##) vs1 vs2›
  and   V_arr_mult[simp]:
            ‹sem_mk_array vs1 * sem_mk_array vs2 = sem_mk_array (map2 (*) vs1 vs2)›
  and   V_arr_sep_disj_R:
            ‹sem_mk_array vs1 ## vs2' ⟹ (∃vs. vs2' = sem_mk_array vs)›
  and   V_named_tup_sep_disj_L:
            ‹vs1' ## sem_mk_array vs2 ⟹ (∃vs. vs1' = sem_mk_array vs)›

lemma sem_mk_array_inj[simp]:
  ‹sem_mk_array vs1 = sem_mk_array vs2 ≡ vs1 = vs2›
  by (smt (verit, best) sem_mk_dest_array)

lemma list_all_replicate[simp]:
  ‹list_all P (replicate n x) ⟷ n = 0 ∨ P x›

lemma 𝖺𝗋𝗋𝖺𝗒_0_eq_any:
  ‹𝖺𝗋𝗋𝖺𝗒[0] A = 𝖺𝗋𝗋𝖺𝗒[0] B›

                    ("𝔸𝗋𝗋𝖺𝗒[_] _" [20, 910] 910)
  where ‹l ⦂ Array N T ≡ sem_mk_array vs ⦂ Itself 𝗌𝗎𝖻𝗃 vs. length l = N ∧ list_all2 (λv x. v ⊨ (x ⦂ T)) vs l›
  deriving ‹Abstract_Domain⇩L T P
        ⟹ Abstract_Domain⇩L (Array N T) (λx. length x = N ∧ list_all P x) ›

        ⟹ Object_Equiv (Array N T) (list_all2 eq)›

       and Functional_Transformation_Functor
       and ‹Is_Aggregate (Array N T)›
    (* and ‹Semantic_Type T TY ⟹ Semantic_Type (Array N T) (𝖺𝗋𝗋𝖺𝗒[N] TY)›

lemma semty_Array [simp, φtype_property Array Semantic_Type]:
  ‹𝗍𝗒𝗉𝖾𝗈𝖿 (Array N T) = 𝖺𝗋𝗋𝖺𝗒[N] (𝗍𝗒𝗉𝖾𝗈𝖿 T)›
  unfolding SType_Of_def Inhabited_def Satisfiable_def Semantic_Type_def
  apply (cases N)
  apply auto
  apply (smt (verit, best) WT_arr Well_Type_unique exE_some less_nat_zero_code list.pred_inject(1) list.size(3) list_all2_conv_all_nth mem_Collect_eq)
  using WT_arr list.pred_inject(1) apply blast
  apply (smt (verit, del_insts) 𝖺𝗋𝗋𝖺𝗒_0_eq_any length_0_conv list.ctr_transfer(1) semty_arr_uniq someI)
  using WT_arr list.pred_inject(1) apply blast
  apply (smt (verit) 𝖺𝗋𝗋𝖺𝗒_0_eq_any exE_some list.size(3) list_all2_Nil2 semty_arr_uniq)
  using WT_arr list.pred_inject(1) apply blast

  subgoal for nat x TY p xa TYa xb
  apply (cases xa; cases xb; auto)
   subgoal premises prems for xx list u lista proof -
      let ?aa = ‹λx v. (∃xa. v = sem_mk_array xa ∧ length x = Suc nat ∧ list_all2 (λv x. v ⊨ (x ⦂ T)) xa x) ›
      let ?xx = ‹λx v TY. ?aa x v ⟶ v ∈ Well_Type TY›
      have t1: ‹∃TY. ∀x v. ?xx x v TY›
        using prems(4) prems(7) by blast
      have t2: ‹?aa (replicate N xx) (sem_mk_array (replicate N u))›
        using list_all2_conv_all_nth prems(1) prems(7) prems(8) by fastforce
      note t3 = someI_ex[OF t1, THEN spec[where x=‹replicate N xx›], THEN spec[where x=‹sem_mk_array (replicate N u)›],
                         THEN mp, OF t2]
      from semty_arr_uniq[OF t3]
        obtain TT where t4: ‹(SOME TY. ∀x v. ?xx x v TY) = 𝖺𝗋𝗋𝖺𝗒[length (replicate N u)] TT›
          by blast
      have t5: ‹sem_mk_array (replicate N u) ∈ Well_Type (𝖺𝗋𝗋𝖺𝗒[length (replicate N u)] TT)›
        using t3 t4 by force
      have t6: ‹u ∈ Well_Type TT›
        using prems(1) t5 by auto
      
      have t7: ‹∃TY.  ∀x v. v ⊨ (x ⦂ T) ⟶ v ∈ Well_Type TY›
        using prems(2) by blast
      from someI_ex[OF t7, THEN spec, THEN spec, THEN mp, OF ‹u ⊨ (xx ⦂ T)›]
        have t8: ‹u ∈ Well_Type (SOME x. ∀xa v. v ⊨ (xa ⦂ T) ⟶ v ∈ Well_Type x)› .
      show ?thesis
        by (smt (verit, del_insts) Eps_cong Well_Type_unique length_replicate prems(1) t2 t4 t6 t8)
      qed .
    
  subgoal premises prems for nat x TY p
    by (insert prems(1-3) prems(4)[THEN spec[where x=‹replicate N x›]],
        metis (no_types, lifting) length_replicate list_all2_conv_all_nth nth_replicate)
  
  subgoal premises prems for nat x TY p
    by (insert prems(1-3) prems(4)[THEN spec[where x=‹𝖺𝗋𝗋𝖺𝗒[N] TY›]],
        auto simp: list_all2_conv_all_nth,
        metis (mono_tags, lifting) list_all_length)
  
    apply (metis (no_types, lifting) length_0_conv list.rel_cases old.nat.distinct(1))
  
  subgoal for nat x TY xa
    apply (cases x; cases xa; auto)
    subgoal premises prems for xx list vv lista
    apply (insert prems(3)[THEN spec[where x=‹replicate N xx›], THEN spec[where x=‹sem_mk_array (replicate N vv)›]], auto)
      using prems(1) apply blast
      apply (simp add: list_all2_conv_all_nth prems(7))
    subgoal premises premx
      apply (insert semty_arr_uniq[OF premx], clarsimp)
    subgoal for TY'
      apply (insert prems(2)[THEN spec[where x=TY']], clarsimp)
    subgoal for x' v'
      apply (insert prems(3)[THEN spec[where x=‹replicate N x'›], THEN spec[where x=‹sem_mk_array (replicate N v')›]],
          auto)
      using prems(1) apply blast
      by (simp add: list_all2_conv_all_nth) . . . .
  .


let_φtype Array
