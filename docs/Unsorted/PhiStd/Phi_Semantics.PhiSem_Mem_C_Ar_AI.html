Theory Phi_Semantics.PhiSem_Mem_C_Ar_AI

theory PhiSem_Mem_C_Ar_AI ― ‹AI stands for ‹arbitrary-precision integer››
  imports PhiSem_Mem_C_Ag_Ar PhiSem_Int_ArbiPrec
begin

section ‹Pointer Arithmetic›

proc op_add_ptr_a[φoverload +]:
  input  ‹i ⦂ 𝗏𝖺𝗅 𝗌𝗅𝗂𝖼𝖾-𝗉𝗍𝗋[addr:len] TY❟ j ⦂ 𝗏𝖺𝗅 ℤ›
  premises ‹0 ≤ int i + j ∧ nat (int i + j) ≤ len ∧ TY ≠ 𝗉𝗈𝗂𝗌𝗈𝗇›
  output ‹nat (int i + j) ⦂ 𝗏𝖺𝗅 𝗌𝗅𝗂𝖼𝖾-𝗉𝗍𝗋[addr:len] TY›
❴
  $i semantic_local_value 𝗉𝗍𝗋
  $j semantic_local_value 𝖺𝗂𝗇𝗍

  semantic_return ‹
      sem_mk_pointer (sem_dest_pointer (φarg.dest 𝖺𝗋𝗀1) ||+ of_int (sem_dest_aint (φarg.dest 𝖺𝗋𝗀2)) * of_nat (MemObj_Size TY))
          ⊨ (nat (int i + j) ⦂ 𝗌𝗅𝗂𝖼𝖾-𝗉𝗍𝗋[addr:len] TY)›
certified proof -
  have t1: ‹address_type addr = 𝖺𝗋𝗋𝖺𝗒[len] TY ∧ TY ≠ 𝗉𝗈𝗂𝗌𝗈𝗇 ∧ (len = 0 ⟶ nat (int i + j) = 0)›
    using the_φ(7) the_φ(8) the_φlemmata(3) by fastforce
  show ?thesis
    by (clarsimp simp: memaddr_to_raw_array_GEP[OF t1] useful distrib_right,
                 simp add: add.commute,
        metis memaddr_to_raw_array_GEP le_zero_eq local.t1 mem_shift_shift of_nat_mult the_φlemmata(1))
  qed
❵ .

proc (nodef) op_add_ptr_aN[φoverload +]:
  input  ‹i ⦂ 𝗏𝖺𝗅 𝗌𝗅𝗂𝖼𝖾-𝗉𝗍𝗋[addr:len] TY❟ j ⦂ 𝗏𝖺𝗅 ℕ›
  premises ‹i + j ≤ len ∧ TY ≠ 𝗉𝗈𝗂𝗌𝗈𝗇›
  output ‹i + j ⦂ 𝗏𝖺𝗅 𝗌𝗅𝗂𝖼𝖾-𝗉𝗍𝗋[addr:len] TY›
❴
  $i + $j
❵ .


lemma nat_int_mul[iff]: "nat (int a * int b) = a * b"
  using nat_times_as_int by presburger
  

end
