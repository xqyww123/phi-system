Theory Phi_Semantics.PhSm_Addr

theory PhSm_Addr
  imports PhiSem_Agg_Void
begin

declare [[typedef_overloaded]]
datatype block = Null | Block nat TY

setup ‹Sign.mandatory_path "block"›

definition ‹layout blk = (case blk of Null ⇒ 𝗏𝗈𝗂𝖽 | Block _ TY ⇒ TY)›

lemma layout[simp]:
  ‹block.layout Null = 𝗏𝗈𝗂𝖽›
  ‹block.layout (Block i TY) = TY›
  unfolding block.layout_def by simp+

setup ‹Sign.parent_path›

(*TODO: rename: block, offset*)
datatype 'index addr = Addr (blk: block) (offset: 'index)
declare [[typedef_overloaded = false]]

declare addr.sel[iff, φsafe_simp]
hide_const (open) blk index

lemma split_memaddr_all: ‹All P ⟷ (∀blk ofs. P (Addr blk ofs))› by (metis addr.exhaust) 
lemma split_memaddr_ex : ‹Ex P  ⟷ (∃blk ofs. P (Addr blk ofs))› by (metis addr.exhaust) 
lemma split_memaddr_meta_all:
  ‹(⋀x. PROP P x) ≡ (⋀blk ofs. PROP P (Addr blk ofs))›
proof
  fix blk ofs
  assume ‹⋀x. PROP P x›
  then show ‹PROP P (Addr blk ofs)› .
next
  fix x
  assume ‹⋀blk ofs. PROP P (Addr blk ofs)›
  note this[of ‹addr.blk x› ‹addr.offset x›, simplified]
  then show ‹PROP P x› .
qed

type_synonym address = "aggregate_path addr" ― ‹Logical address›

subsubsection ‹Algebraic Properties›

instantiation block :: zero begin
definition [simp]: "zero_block = Null"
instance ..
end

instantiation addr :: (zero) zero begin
definition "zero_addr = (Addr 0 0)"
instance ..
end

lemma memaddr_blk_zero[simp]:
  ‹addr.blk 0 = Null›
  unfolding zero_addr_def by simp

lemma memaddr_ofs_zero[simp]:
  ‹addr.offset 0 = 0›
  unfolding zero_addr_def by simp

paragraph ‹Freshness›

lemma infinite_UNIV_int [simp]: "infinite (UNIV::int set)"
proof
  assume "finite (UNIV::int set)"
  moreover have "inj (λi::int. 2 * i)"
    by (rule injI) simp
  ultimately have "surj (λi::int. 2 * i)"
    by (rule finite_UNIV_inj_surj)
  thm finite_UNIV_inj_surj
  then obtain i :: int where "1 = 2 * i" by (rule surjE)
  then show False by presburger
qed

lemma block_infinite[simp]:
  ‹infinite (UNIV :: block set)›
  using inj_on_finite[where A = ‹UNIV::nat set› and B = ‹(UNIV :: block set)›
        and f = ‹λn. Block n undefined›]
  by (meson infinite_UNIV_char_0 injI block.inject top_greatest)

lemma block_infinite_TY:
  ‹infinite {a. block.layout a = TY}›
  using inj_on_finite[where A = ‹UNIV::nat set› and B = ‹{a. block.layout a = TY}›
        and f = ‹λn. Block n TY›]
  using inj_def by fastforce

subsection ‹Address Type›

definition address_type :: ‹address ⇒ TY›
  where ‹address_type addr ≡ index_type (addr.offset addr) (block.layout (addr.blk addr))›

adhoc_overloading Type_Of_syntax address_type
