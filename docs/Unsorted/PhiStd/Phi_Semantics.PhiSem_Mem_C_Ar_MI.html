section ‹Pointer Arithmetic›

proc op_add_ptr[φoverload +]:
  input  ‹i ⦂ 𝗏𝖺𝗅 𝗌𝗅𝗂𝖼𝖾-𝗉𝗍𝗋[addr:len] TY❟ j ⦂ 𝗏𝖺𝗅 ℤ('b)›
  premises ‹0 ≤ int i + j ∧ nat (int i + j) ≤ len ∧ TY ≠ 𝗉𝗈𝗂𝗌𝗈𝗇›
  output ‹nat (int i + j) ⦂ 𝗏𝖺𝗅 𝗌𝗅𝗂𝖼𝖾-𝗉𝗍𝗋[addr:len] TY›
❴
  $i semantic_local_value 𝗉𝗍𝗋
  $j semantic_local_value ‹𝗂𝗇𝗍('b)› 

  have [simp]: ‹word_of_nat (nat (2 ^ LENGTH('b) + j)) = (word_of_int j :: 'b word)›
    by (smt (verit, ccfv_threshold) One_nat_def nat_0_le of_int_of_nat_eq sint_of_int_eq the_φlemmata(1) two_less_eq_exp_length wi_hom_add word_of_int_0 word_of_int_2p_len) ；

  semantic_return ‹
      sem_mk_pointer (sem_dest_pointer (φarg.dest 𝖺𝗋𝗀1)
              ||+ Word.scast (of_nat (snd (sem_dest_int (φarg.dest 𝖺𝗋𝗀2))) :: 'b word) * of_nat (MemObj_Size TY))
          ⊨ (nat (int i + j) ⦂ 𝗌𝗅𝗂𝖼𝖾-𝗉𝗍𝗋[addr:len] TY)›
  certified proof -
    have t1: ‹address_type addr = 𝖺𝗋𝗋𝖺𝗒[len] TY ∧ TY ≠ 𝗉𝗈𝗂𝗌𝗈𝗇 ∧ (len = 0 ⟶ nat (int i + j) = 0) ›
      by auto_sledgehammer
    show ?thesis
      by (insert useful, auto simp: memaddr_to_raw_array_GEP[OF t1] distrib_right,
                  simp add: add.commute signed_of_int signed_take_bit_int_eq_self;
                  auto_sledgehammer)
  qed
❵ .


proc op_add_ptr_unsigned[φoverload +]:
  input  ‹i ⦂ 𝗏𝖺𝗅 𝗌𝗅𝗂𝖼𝖾-𝗉𝗍𝗋[addr:len] TY❟ j ⦂ 𝗏𝖺𝗅 ℕ('b)›
  premises ‹i + j ≤ len ∧ TY ≠ 𝗉𝗈𝗂𝗌𝗈𝗇›
  output ‹i + j ⦂ 𝗏𝖺𝗅 𝗌𝗅𝗂𝖼𝖾-𝗉𝗍𝗋[addr:len] TY›
❴
  $i semantic_local_value 𝗉𝗍𝗋
  $j semantic_local_value ‹𝗂𝗇𝗍('b)›

  semantic_return ‹
      sem_mk_pointer (sem_dest_pointer (φarg.dest 𝖺𝗋𝗀1)
              ||+ Word.ucast (of_nat (snd (sem_dest_int (φarg.dest 𝖺𝗋𝗀2))) :: 'b word) * of_nat (MemObj_Size TY))
          ⊨ (i + j ⦂ 𝗌𝗅𝗂𝖼𝖾-𝗉𝗍𝗋[addr:len] TY)›
  certified proof -
    have t1: ‹address_type addr = 𝖺𝗋𝗋𝖺𝗒[len] TY ∧ TY ≠ 𝗉𝗈𝗂𝗌𝗈𝗇 ∧ (len = 0 ⟶ i + j = 0) ›
      by auto_sledgehammer
    show ?thesis
      by (insert useful,
                  auto simp: memaddr_to_raw_array_GEP[OF t1]
                             distrib_right ucast_of_nat_small; auto_sledgehammer)
    qed
❵ .



declare nat_int_add[iff]
