Theory PhiStd_Slice_a

theory PhiStd_Slice_a ― ‹'a' for arbitary precision interger›
  imports PhiStd_Loop_a
          Phi_Semantics.PhiSem_Mem_C_Ar_AI
begin

text ‹Predefined abstractions of Loop statemetns, counted as a part of loop invariants in our statistics›

proc (nodef) map_slice_a:
  requires body: ‹⋀k v. 𝖼𝗈𝗇𝖽𝗂𝗍𝗂𝗈𝗇 k < length l
                     ⟹ 𝗉𝗋𝗈𝖼 Body v ⦃ k ⦂ 𝗏𝖺𝗅[v] ℕ❟ l ! k ⦂ 𝗆𝖾𝗆[addr ‣ (i+k)⇧𝗍⇧𝗁] T❟ X
                                  ⟼ f k (l ! k) ⦂ 𝗆𝖾𝗆[addr ‣ (i+k)⇧𝗍⇧𝗁] T❟ X ⦄›

  input  ‹len ⦂ 𝗏𝖺𝗅 ℕ❟ l ⦂ 𝗆𝖾𝗆[addr] 𝗌𝗅𝗂𝖼𝖾[i,len] T❟ X›
  output ‹map_index f l ⦂ 𝗆𝖾𝗆[addr] 𝗌𝗅𝗂𝖼𝖾[i,len] T❟ X›
  for T :: ‹(mem_fic, 'a) φ›
❴
  note ‹length l = len› [simp] ；
  map_list_loop_a ($len) ‹λj. 𝗆𝖾𝗆[addr ‣ (i+j)⇧𝗍⇧𝗁] T› ❴ for j
    body
  ❵
❵ .

proc (nodef) map_2slice_a:
  requires body: ‹⋀k v. 𝖼𝗈𝗇𝖽𝗂𝗍𝗂𝗈𝗇 k < len
                     ⟹ 𝗉𝗋𝗈𝖼 Body v ⦃ k ⦂ 𝗏𝖺𝗅[v] ℕ❟
                                       l⇩a ! k ⦂ 𝗆𝖾𝗆[addr⇩a ‣ (i⇩a+k)⇧𝗍⇧𝗁] T⇩a❟
                                       l⇩b ! k ⦂ 𝗆𝖾𝗆[addr⇩b ‣ (i⇩b+k)⇧𝗍⇧𝗁] T⇩b❟ X
                                  ⟼ f k (l⇩b ! k) (l⇩a ! k) ⦂ 𝗆𝖾𝗆[addr⇩a ‣ (i⇩a+k)⇧𝗍⇧𝗁] T⇩a❟
                                      l⇩b ! k ⦂ 𝗆𝖾𝗆[addr⇩b ‣ (i⇩b+k)⇧𝗍⇧𝗁] T⇩b❟ X ⦄›

  input  ‹len ⦂ 𝗏𝖺𝗅 ℕ❟ l⇩a ⦂ 𝗆𝖾𝗆[addr⇩a] 𝗌𝗅𝗂𝖼𝖾[i⇩a,len] T⇩a❟ l⇩b ⦂ 𝗆𝖾𝗆[addr⇩b] 𝗌𝗅𝗂𝖼𝖾[i⇩b,len] T⇩b❟ X›
  output ‹map_index (λi (a,b). f i b a) (zip l⇩a l⇩b) ⦂ 𝗆𝖾𝗆[addr⇩a] 𝗌𝗅𝗂𝖼𝖾[i⇩a,len] T⇩a❟
          l⇩b ⦂ 𝗆𝖾𝗆[addr⇩b] 𝗌𝗅𝗂𝖼𝖾[i⇩b,len] T⇩b❟ X›
  for T⇩a :: ‹(mem_fic, 'a) φ›
  and T⇩b :: ‹(mem_fic, 'b) φ›
❴
  note ‹length l⇩a = len› [simp] ‹length l⇩b = len› [simp] ；
  map_2list_loop_a ($len) ‹(λj. 𝗆𝖾𝗆[addr⇩a ‣ (i⇩a+j)⇧𝗍⇧𝗁] T⇩a, λj. 𝗆𝖾𝗆[addr⇩b ‣ (i⇩b+j)⇧𝗍⇧𝗁] T⇩b)› ❴
    body 
  ❵
❵ .


proc memcpy_a:
  requires ‹(⋀x. Semantic_Type T TY)›
  input  ‹i⇩a ⦂ 𝗏𝖺𝗅 𝗌𝗅𝗂𝖼𝖾-𝗉𝗍𝗋[addr⇩a:LEN⇩a] TY❟ i⇩b ⦂ 𝗏𝖺𝗅 𝗌𝗅𝗂𝖼𝖾-𝗉𝗍𝗋[addr⇩b:LEN⇩b] TY❟ len ⦂ 𝗏𝖺𝗅 ℕ❟
          l⇩a ⦂ 𝗆𝖾𝗆[addr⇩a] 𝗌𝗅𝗂𝖼𝖾[i⇩a,len] T❟ l⇩b ⦂ 𝗆𝖾𝗆[addr⇩b] 𝗌𝗅𝗂𝖼𝖾[i⇩b,len] T›
  premises ‹i⇩a + len ≤ LEN⇩a ∧ i⇩b + len ≤ LEN⇩b›
  output ‹l⇩b ⦂ 𝗆𝖾𝗆[addr⇩a] 𝗌𝗅𝗂𝖼𝖾[i⇩a,len] T❟ l⇩b ⦂ 𝗆𝖾𝗆[addr⇩b] 𝗌𝗅𝗂𝖼𝖾[i⇩b,len] T›
  for T :: ‹(VAL, 'x) φ›
❴
  map_2slice_a ($len) ❴ → val k
     $i⇩a + $k := ($i⇩b + $k) !
  ❵
❵ .


end
